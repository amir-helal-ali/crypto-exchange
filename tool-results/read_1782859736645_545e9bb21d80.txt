    74тЖТ    74тЖТ// Returns information about the currently configured SSL certificate.
    75тЖТ    75тЖТ// Parses the cert file (if exists) using Go's crypto/x509 тАФ no shell-out.
    76тЖТ    76тЖТ//
    77тЖТ    77тЖТ// GET /api/v1/admin/ssl/status
    78тЖТ    78тЖТfunc GetSSLStatus(c *gin.Context) {
    79тЖТ    79тЖТ        certPath := settings.Get("ssl_cert_path")
    80тЖТ    80тЖТ        keyPath := settings.Get("ssl_key_path")
    81тЖТ    81тЖТ        sslEnabled := settings.GetBool("ssl_enabled")
    82тЖТ    82тЖТ        sslType := settings.GetDefault("ssl_type", "unknown")
    83тЖТ    83тЖТ
    84тЖТ    84тЖТ        response := gin.H{
    85тЖТ    85тЖТ                "enabled":      sslEnabled,
    86тЖТ    86тЖТ                "type":         sslType,
    87тЖТ    87тЖТ                "issuer":       settings.GetDefault("ssl_issuer", "тАФ"),
    88тЖТ    88тЖТ                "cert_path":    certPath,
    89тЖТ    89тЖТ                "key_path":     keyPath,
    90тЖТ    90тЖТ                "domains":      settings.Get("ssl_domains"),
    91тЖТ    91тЖТ                "email":        settings.Get("ssl_email"),
    92тЖТ    92тЖТ                "generated_at": settings.Get("ssl_generated_at"),
    93тЖТ    93тЖТ                "exists":       false,
    94тЖТ    94тЖТ        }
    95тЖТ    95тЖТ
    96тЖТ    96тЖТ        if certPath == "" {
    97тЖТ    97тЖТ                c.JSON(200, gin.H{"success": true, "data": response})
    98тЖТ    98тЖТ                return
    99тЖТ    99тЖТ        }
   100тЖТ   100тЖТ
   101тЖТ   101тЖТ        // Try to read + parse the cert file
   102тЖТ   102тЖТ        certBytes, err := os.ReadFile(certPath)
   103тЖТ   103тЖТ        if err != nil {
   104тЖТ   104тЖТ                response["error"] = fmt.Sprintf("cert file not readable: %v", err)
   105тЖТ   105тЖТ                c.JSON(200, gin.H{"success": true, "data": response})
   106тЖТ   106тЖТ                return
   107тЖТ   107тЖТ        }
   108тЖТ   108тЖТ
   109тЖТ   109тЖТ        block, _ := pem.Decode(certBytes)
   110тЖТ   110тЖТ        if block == nil {
   111тЖТ   111тЖТ                response["error"] = "cert file is not valid PEM"
   112тЖТ   112тЖТ                c.JSON(200, gin.H{"success": true, "data": response})
   113тЖТ   113тЖТ                return
   114тЖТ   114тЖТ        }
   115тЖТ   115тЖТ
   116тЖТ   116тЖТ        cert, err := x509.ParseCertificate(block.Bytes)
   117тЖТ   117тЖТ        if err != nil {
   118тЖТ   118тЖТ                response["error"] = fmt.Sprintf("cert parse failed: %v", err)
   119тЖТ   119тЖТ                c.JSON(200, gin.H{"success": true, "data": response})
   120тЖТ   120тЖТ                return
   121тЖТ   121тЖТ        }
   122тЖТ   122тЖТ
   123тЖТ   123тЖТ        // Calculate days remaining
   124тЖТ   124тЖТ        daysRemaining := int(time.Until(cert.NotAfter).Hours() / 24)
   125тЖТ   125тЖТ
   126тЖТ   126тЖТ        // Health check
   127тЖТ   127тЖТ        health := "healthy"
   128тЖТ   128тЖТ        if daysRemaining < 0 {
   129тЖТ   129тЖТ                health = "expired"
   130тЖТ   130тЖТ        } else if daysRemaining < 7 {
   131тЖТ   131тЖТ                health = "critical"
   132тЖТ   132тЖТ        } else if daysRemaining < 30 {
   133тЖТ   133тЖТ                health = "warning"
   134тЖТ   134тЖТ        }
   135тЖТ   135тЖТ
   136тЖТ   136тЖТ        response["exists"] = true
   137тЖТ   137тЖТ        response["subject"] = cert.Subject.CommonName
   138тЖТ   138тЖТ        response["issuer"] = cert.Issuer.CommonName
   139тЖТ   139тЖТ        response["issuer_org"] = strings.Join(cert.Issuer.Organization, ", ")
   140тЖТ   140тЖТ        response["not_before"] = cert.NotBefore.Format(time.RFC3339)
   141тЖТ   141тЖТ        response["not_after"] = cert.NotAfter.Format(time.RFC3339)
   142тЖТ   142тЖТ        response["expires_at"] = cert.NotAfter.Format("2006-01-02")
   143тЖТ   143тЖТ        response["days_remaining"] = daysRemaining
   144тЖТ   144тЖТ        response["health"] = health
   145тЖТ   145тЖТ        response["sans"] = cert.DNSNames
   146тЖТ   146тЖТ        response["key_algorithm"] = cert.PublicKeyAlgorithm.String()
   147тЖТ   147тЖТ        response["signature_algorithm"] = cert.SignatureAlgorithm.String()
   148тЖТ   148тЖТ        response["serial_number"] = cert.SerialNumber.String()
   149тЖТ   149тЖТ
   150тЖТ   150тЖТ        // Check key file too
   151тЖТ   151тЖТ        if _, err := os.Stat(keyPath); err != nil {
   152тЖТ   152тЖТ                response["key_exists"] = false
   153тЖТ   153тЖТ                response["error"] = "key file not found"
   154тЖТ   154тЖТ        } else {
   155тЖТ   155тЖТ                response["key_exists"] = true
   156тЖТ   156тЖТ        }
   157тЖТ   157тЖТ
   158тЖТ   158тЖТ        c.JSON(200, gin.H{"success": true, "data": response})
   159тЖТ   159тЖТ}

   161тЖТ   161тЖТ// тФАтФАтФА GenerateSSLCertificate тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   162тЖТ   162тЖТ// One-click SSL cert generation. Auto-detects domains from settings.
   163тЖТ   163тЖТ//
   164тЖТ   164тЖТ// POST /api/v1/admin/ssl/generate
   165тЖТ   165тЖТ// Body: {
   166тЖТ   166тЖТ//   "type": "local" | "letsencrypt",
   167тЖТ   167тЖТ//   "domains": ["eg-money.local", ...],  // optional, auto-filled from settings
   168тЖТ   168тЖТ//   "email": "admin@example.com",         // required for letsencrypt
   169тЖТ   169тЖТ//   "staging": false                      // use LE staging env (for testing)
   170тЖТ   170тЖТ// }
   171тЖТ   171тЖТfunc GenerateSSLCertificate(c *gin.Context) {
   172тЖТ   172тЖТ        var req generateSSLRequest
   173тЖТ   173тЖТ        if err := c.ShouldBindJSON(&req); err != nil {
   174тЖТ   174тЖТ                c.JSON(400, gin.H{"error": "Invalid request", "details": err.Error()})
   175тЖТ   175тЖТ                return
   176тЖТ   176тЖТ        }
   177тЖТ   177тЖТ
   178тЖТ   178тЖТ        // Normalize type
   179тЖТ   179тЖТ        req.Type = strings.ToLower(strings.TrimSpace(req.Type))
   180тЖТ   180тЖТ        if req.Type != "local" && req.Type != "letsencrypt" {
   181тЖТ   181тЖТ                c.JSON(400, gin.H{"error": "type must be 'local' or 'letsencrypt'"})
   182тЖТ   182тЖТ                return
   183тЖТ   183тЖТ        }
   184тЖТ   184тЖТ
   185тЖТ   185тЖТ        // Auto-detect domains from settings if not provided
   186тЖТ   186тЖТ        if len(req.Domains) == 0 {
   187тЖТ   187тЖТ                req.Domains = autoDetectDomains()
   188тЖТ   188тЖТ        }
   189тЖТ   189тЖТ        if len(req.Domains) == 0 {
   190тЖТ   190тЖТ                c.JSON(400, gin.H{"error": "No domains configured. Set domains in the Domains tab first."})
   191тЖТ   191тЖТ                return
   192тЖТ   192тЖТ        }
   193тЖТ   193тЖТ
   194тЖТ   194тЖТ        // Sanitize domains (lowercase, trim, dedupe)
   195тЖТ   195тЖТ        req.Domains = sanitizeDomains(req.Domains)
   196тЖТ   196тЖТ        if len(req.Domains) == 0 {
   197тЖТ   197тЖТ                c.JSON(400, gin.H{"error": "No valid domains provided"})
   198тЖТ   198тЖТ                return
   199тЖТ   199тЖТ        }
   200тЖТ   200тЖТ
   201тЖТ   201тЖТ        // Validate email for letsencrypt
   202тЖТ   202тЖТ        if req.Type == "letsencrypt" && req.Email == "" {
   203тЖТ   203тЖТ                c.JSON(400, gin.H{"error": "Email is required for Let's Encrypt certificates"})
   204тЖТ   204тЖТ                return
   205тЖТ   205тЖТ        }
   206тЖТ   206тЖТ
   207тЖТ   207тЖТ        // Validate domains for letsencrypt (must be real, public-facing domains)
   208тЖТ   208тЖТ        if req.Type == "letsencrypt" {
   209тЖТ   209тЖТ                for _, d := range req.Domains {
   210тЖТ   210тЖТ                        if isLocalDomain(d) {
   211тЖТ   211тЖТ                                c.JSON(400, gin.H{
   212тЖТ   212тЖТ                                        "error": fmt.Sprintf("Let's Encrypt cannot issue certs for local domain '%s'. Use 'local' type for development.", d),
   213тЖТ   213тЖТ                                })
   214тЖТ   214тЖТ                                return
   215тЖТ   215тЖТ                        }
   216тЖТ   216тЖТ                }
   217тЖТ   217тЖТ        }
   218тЖТ   218тЖТ
   219тЖТ   219тЖТ        // Ensure certs directory exists
   220тЖТ   220тЖТ        if err := os.MkdirAll(certsDir(), 0755); err != nil {
   221тЖТ   221тЖТ                c.JSON(500, gin.H{"error": "Cannot create certs directory", "details": err.Error()})
   222тЖТ   222тЖТ                return
   223тЖТ   223тЖТ        }
   224тЖТ   224тЖТ
   225тЖТ   225тЖТ        // Determine cert file paths. Use a single wildcard-style name based on
   226тЖТ   226тЖТ        // the primary domain so it's easy to find later.
   227тЖТ   227тЖТ        primaryDomain := req.Domains[0]
   228тЖТ   228тЖТ        certPath := filepath.Join(certsDir(), primaryDomain+".pem")
   229тЖТ   229тЖТ        keyPath := filepath.Join(certsDir(), primaryDomain+"-key.pem")
   230тЖТ   230тЖТ
   231тЖТ   231тЖТ        // Generate the cert
   232тЖТ   232тЖТ        var output strings.Builder
   233тЖТ   233тЖТ        var err error
   234тЖТ   234тЖТ
   235тЖТ   235тЖТ        switch req.Type {
   236тЖТ   236тЖТ        case "local":
   237тЖТ   237тЖТ                output, err = generateLocalCert(req.Domains, certPath, keyPath)
   238тЖТ   238тЖТ        case "letsencrypt":
   239тЖТ   239тЖТ                output, err = generateLetsEncryptCert(req.Domains, req.Email, req.Staging, certPath, keyPath)
   240тЖТ   240тЖТ        }
   241тЖТ   241тЖТ
   242тЖТ   242тЖТ        if err != nil {
   243тЖТ   243тЖТ                c.JSON(500, gin.H{
   244тЖТ   244тЖТ                        "error":   "Certificate generation failed",
   245тЖТ   245тЖТ                        "details": err.Error(),
   246тЖТ   246тЖТ                        "output":  output.String(),
   247тЖТ   247тЖТ                })
   248тЖТ   248тЖТ                return
   249тЖТ   249тЖТ        }
   250тЖТ   250тЖТ
   251тЖТ   251тЖТ        // Update DB settings
   252тЖТ   252тЖТ        userID, _ := c.Get("user_id")
   253тЖТ   253тЖТ        uid, _ := userID.(uint)
   254тЖТ   254тЖТ
   255тЖТ   255тЖТ        issuer := "Self-Signed (OpenSSL)"
   256тЖТ   256тЖТ        if req.Type == "letsencrypt" {
   257тЖТ   257тЖТ                if req.Staging {
   258тЖТ   258тЖТ                        issuer = "Let's Encrypt (Staging)"
   259тЖТ   259тЖТ                } else {
   260тЖТ   260тЖТ                        issuer = "Let's Encrypt"
   261тЖТ   261тЖТ                }
   262тЖТ   262тЖТ        }
   263тЖТ   263тЖТ
   264тЖТ   264тЖТ        now := time.Now().UTC()
   265тЖТ   265тЖТ        settings.Set("ssl_cert_path", certPath, "ssl", uid)
   266тЖТ   266тЖТ        settings.Set("ssl_key_path", keyPath, "ssl", uid)
   267тЖТ   267тЖТ        settings.Set("ssl_enabled", "true", "ssl", uid)
   268тЖТ   268тЖТ        settings.Set("ssl_type", req.Type, "ssl", uid)
   269тЖТ   269тЖТ        settings.Set("ssl_issuer", issuer, "ssl", uid)
   270тЖТ   270тЖТ        settings.Set("ssl_generated_at", now.Format(time.RFC3339), "ssl", uid)
   271тЖТ   271тЖТ        settings.Set("ssl_domains", strings.Join(req.Domains, ","), "ssl", uid)
   272тЖТ   272тЖТ        if req.Email != "" {
   273тЖТ   273тЖТ                settings.Set("ssl_email", req.Email, "ssl", uid)
   274тЖТ   274тЖТ        }
   275тЖТ   275тЖТ        settings.Set("ssl_staging", boolStr(req.Staging), "ssl", uid)
   276тЖТ   276тЖТ
   277тЖТ   277тЖТ        // Read cert expiry
   278тЖТ   278тЖТ        if expiry, err := readCertExpiry(certPath); err == nil {
   279тЖТ   279тЖТ                settings.Set("ssl_expiry", expiry.Format(time.RFC3339), "ssl", uid)
   280тЖТ   280тЖТ        }
   281тЖТ   281тЖТ
   282тЖТ   282тЖТ        // Log to audit
   283тЖТ   283тЖТ        database.DB.Create(&models.AuditLog{
   284тЖТ   284тЖТ                UserID:    uid,
   285тЖТ   285тЖТ                Action:    "SSL_CERT_GENERATED",
   286тЖТ   286тЖТ                Details:   fmt.Sprintf("Generated %s SSL cert for: %s (issuer: %s)", req.Type, strings.Join(req.Domains, ", "), issuer),
   287тЖТ   287тЖТ                IPAddress: c.ClientIP(),
   288тЖТ   288тЖТ        })
   289тЖТ   289тЖТ
   290тЖТ   290тЖТ        // Trigger nginx reload
   291тЖТ   291тЖТ        triggerNginxReload()
   292тЖТ   292тЖТ
   293тЖТ   293тЖТ        c.JSON(200, gin.H{
   294тЖТ   294тЖТ                "success":   true,
   295тЖТ   295тЖТ                "message":   "SSL certificate generated successfully. Nginx is reloading.",
   296тЖТ   296тЖТ                "type":      req.Type,
   297тЖТ   297тЖТ                "issuer":    issuer,
   298тЖТ   298тЖТ                "domains":   req.Domains,
   299тЖТ   299тЖТ                "cert_path": certPath,
   300тЖТ   300тЖТ                "key_path":  keyPath,
   301тЖТ   301тЖТ                "output":    output.String(),
   302тЖТ   302тЖТ        })
   303тЖТ   303тЖТ}

   483тЖТ   483тЖТ// generateLocalCert creates a self-signed cert using openssl with proper SANs.
   484тЖТ   484тЖТ// The cert is valid for 365 days and includes all provided domains as SANs.
   485тЖТ   485тЖТ//
   486тЖТ   486тЖТ// Note: Browsers will show a "not trusted" warning for self-signed certs.
   487тЖТ   487тЖТ// For trusted local certs, the admin should install mkcert in the backend
   488тЖТ   488тЖТ// container (see Dockerfile comments) тАФ but self-signed is the universal
   489тЖТ   489тЖТ// fallback that works everywhere.
   490тЖТ   490тЖТfunc generateLocalCert(domains []string, certPath, keyPath string) (strings.Builder, error) {
   491тЖТ   491тЖТ        var output strings.Builder
   492тЖТ   492тЖТ
   493тЖТ   493тЖТ        // Build SAN config
   494тЖТ   494тЖТ        sanEntries := []string{}
   495тЖТ   495тЖТ        for _, d := range domains {
   496тЖТ   496тЖТ                sanEntries = append(sanEntries, "DNS:"+d)
   497тЖТ   497тЖТ        }
   498тЖТ   498тЖТ        sanList := strings.Join(sanEntries, ",")
   499тЖТ   499тЖТ
   500тЖТ   500тЖТ        // Write a temp openssl config with SAN extension
   501тЖТ   501тЖТ        tmpConfig, err := os.CreateTemp("", "openssl-san-*.cnf")
   502тЖТ   502тЖТ        if err != nil {
   503тЖТ   503тЖТ                return output, fmt.Errorf("create temp config: %w", err)
   504тЖТ   504тЖТ        }
   505тЖТ   505тЖТ        defer os.Remove(tmpConfig.Name())
   506тЖТ   506тЖТ
   507тЖТ   507тЖТ        configContent := fmt.Sprintf(`[req]
   508тЖТ   508тЖТdistinguished_name = req_distinguished_name

   510тЖТ   510тЖТprompt = no
   511тЖТ   511тЖТ
   512тЖТ   512тЖТ[req_distinguished_name]
   513тЖТ   513тЖТCN = %s

   530тЖТ   530тЖТ                tmpConfig.Close()

   557тЖТ   557тЖТ                output.WriteString(fmt.Sprintf("\n[warn] chmod cert failed: %v", err))

   561тЖТ   561тЖТ        output.WriteString(fmt.Sprintf("[ok] Cert: %s\n", certPath))
   562тЖТ   562тЖТ        output.WriteString(fmt.Sprintf("[ok] Key:  %s\n", keyPath))
   563тЖТ   563тЖТ

   686тЖТ   686тЖТ        // IP address regex

   688тЖТ   688тЖТ                return true
   689тЖТ   689тЖТ        }
   690тЖТ   690тЖТ        return false
