     4→import { useRouter, usePathname } from "next/navigation";
     5→import Link from "next/link";
     6→import toast from "react-hot-toast";
     7→import {
     8→  LayoutDashboard,
     9→  TrendingUp,

    35→import { authGet, authPut } from "@/lib/api";
    36→
    37→/* ─────────── Navigation Config ─────────── */
    38→const NAV_SECTIONS = [
    39→  {
    40→    items: [
    41→      { href: "/dashboard", icon: LayoutDashboard, label: "لوحة التحكم" },
    42→      { href: "/dashboard/exchange", icon: TrendingUp, label: "التداول" },

    58→const TICKER_SYMBOLS = [
    59→  { symbol: "BTCUSDT", label: "BTC", color: "text-amber-400" },
    60→  { symbol: "ETHUSDT", label: "ETH", color: "text-blue-400" },
    61→  { symbol: "BNBUSDT", label: "BNB", color: "text-yellow-400" },
    62→  { symbol: "SOLUSDT", label: "SOL", color: "text-purple-400" },
    63→];
    64→
    65→/* ─────────── Types ─────────── */
    66→interface Notification {
    67→  id: number;
    68→  type: string;
    69→  title: string;
    70→  body: string;
    71→  data: string;
    72→  read: boolean;
    73→  created_at: string;
    74→}
    75→
    76→/* ─────────── Icon Map (lucide-react) ─────────── */
    77→const NOTIF_ICON_MAP: Record<string, { icon: any; bg: string; fg: string }> = {
    78→  ORDER_FILLED: { icon: Check, bg: "bg-emerald-500/15", fg: "text-emerald-400" },
    79→  ORDER_CANCELLED: { icon: Ban, bg: "bg-red-500/15", fg: "text-red-400" },
    80→  ORDER_TRIGGERED: { icon: Zap, bg: "bg-amber-500/15", fg: "text-amber-400" },
    81→  DEPOSIT_APPROVED: { icon: Banknote, bg: "bg-emerald-500/15", fg: "text-emerald-400" },
    82→  WITHDRAWAL_APPROVED: { icon: SendHorizonal, bg: "bg-teal-500/15", fg: "text-teal-400" },
    83→  WITHDRAWAL_REJECTED: { icon: AlertTriangle, bg: "bg-red-500/15", fg: "text-red-400" },
    84→  KYC_APPROVED: { icon: ShieldCheck, bg: "bg-emerald-500/15", fg: "text-emerald-400" },
    85→  KYC_REJECTED: { icon: AlertTriangle, bg: "bg-orange-500/15", fg: "text-orange-400" },
    86→};
    87→const DEFAULT_NOTIF_ICON = { icon: Bell, bg: "bg-primary/15", fg: "text-primary" };
    88→
    89→/* ─────────── Component ─────────── */
    90→export default function DashboardLayout({ children }: { children: React.ReactNode }) {
    91→  const router = useRouter();
    92→  const pathname = usePathname();
    93→  const [user, setUser] = useState<any>(null);
    94→  const [sidebarOpen, setSidebarOpen] = useState(false);
    95→  const [prices, setPrices] = useState<Record<string, { price: string; change: number }>>({});
    96→  const [prevPrices, setPrevPrices] = useState<Record<string, string>>({});
    97→  const [userMenuOpen, setUserMenuOpen] = useState(false);
    98→  const [notifOpen, setNotifOpen] = useState(false);
    99→  const [notifications, setNotifications] = useState<Notification[]>([]);
   100→  const [unreadCount, setUnreadCount] = useState(0);
   101→  const menuRef = useRef<HTMLDivElement>(null);
   102→  const notifRef = useRef<HTMLDivElement>(null);
   103→
   104→  /* ── Auth & Price WebSocket ── */
   105→  useEffect(() => {
   106→    const token = localStorage.getItem("token");
   107→    if (!token) { router.push("/login"); return; }
   108→    try { setUser(JSON.parse(localStorage.getItem("user") || "{}")); } catch { router.push("/login"); }
   109→
   110→    const ws = new WebSocket("wss://stream.binance.com:9443/ws/!miniTicker@arr");
   111→    ws.onmessage = (e) => {
   112→      const data = JSON.parse(e.data);
   113→      if (Array.isArray(data)) {
   114→        const updates: Record<string, { price: string; change: number }> = {};
   115→        const newPrev: Record<string, string> = {};
   116→        data.forEach((t: any) => {
   117→          if (["BTCUSDT", "ETHUSDT", "BNBUSDT", "SOLUSDT"].includes(t.s)) {
   118→            const prev = prevPrices[t.s];
   119→            const change = prev ? (parseFloat(t.c) - parseFloat(prev)) / parseFloat(prev) * 100 : 0;
   120→            updates[t.s] = { price: t.c, change };
   121→            newPrev[t.s] = t.c;
   122→          }
   123→        });
   124→        setPrices(prev => {
   125→          const merged = { ...prev };
   126→          Object.keys(updates).forEach(k => {
   127→            merged[k] = { ...prev[k], ...updates[k] };
   128→          });
   129→          return merged;
   130→        });
   131→        setPrevPrices(p => ({ ...p, ...newPrev }));
   132→      }
   133→    };
   134→    return () => ws.close();
   135→  }, [router]);
   136→
   137→  /* ── Notifications + User WebSocket ── */
   138→  useEffect(() => {
   139→    const fetchNotifs = async () => {
   140→      try {
   141→        const res = await authGet("/api/v1/notifications?limit=20");
   142→        if (res.ok) {
   143→          const data = await res.json();
   144→          setNotifications(data.data || []);
   145→          setUnreadCount(data.unread_count || 0);
   146→        }
   147→      } catch {}
   148→    };
   149→    fetchNotifs();
   150→    const interval = setInterval(fetchNotifs, 30000);
   151→
   152→    const token = localStorage.getItem("token");
   153→    const apiUrl = process.env.NEXT_PUBLIC_API_URL || "http://localhost:3000";
   154→    const wsUrl = apiUrl.replace(/^http/, "ws") + "/ws/user?token=" + token;
   155→    let userWs: WebSocket | null = null;
   156→    let reconnectTimer: number;
   157→
   158→    const connectUserWs = () => {
   159→      try {
   160→        userWs = new WebSocket(wsUrl);
   161→        userWs.onmessage = (e) => {
   162→          try {
   163→            const event = JSON.parse(e.data);
   164→            if (event.type === "order_fill") {
   165→              toast.success(`تم تنفيذ الأمر: ${event.data.side === "BUY" ? "شراء" : "بيع"} ${event.data.quantity} ${event.data.symbol}`);
   166→            } else if (event.type === "order_trigger") {
   167→              toast(`تم تفعيل الأمر الشرطي: ${event.data.symbol}`, { icon: "⚡" });
   168→            } else if (event.type === "deposit_approved") {
   169→              toast.success(`تم تأكيد الإيداع: ${event.data.amount} ${event.data.currency}`);
   170→            } else if (event.type === "withdrawal_approved") {
   171→              toast.success(`تم تأكيد السحب: ${event.data.amount} ${event.data.currency}`);
   172→            } else if (event.type === "withdrawal_rejected") {
   173→              toast.error(`تم رفض السحب: ${event.data.amount} ${event.data.currency}`);
   174→            } else if (event.type === "kyc_APPROVED") {
   175→              toast.success("تمت الموافقة على التحقق من الهوية");
   176→            } else if (event.type === "kyc_REJECTED") {
   177→              toast.error("تم رفض طلب التحقق من الهوية");
   178→            }
   179→            fetchNotifs();
   180→          } catch {}
   181→        };
   182→        userWs.onclose = () => { reconnectTimer = window.setTimeout(connectUserWs, 5000); };
   183→        userWs.onerror = () => { userWs?.close(); };
   184→      } catch {}
   185→    };
   186→    connectUserWs();
   187→
   188→    return () => {
   189→      clearInterval(interval);
   190→      clearTimeout(reconnectTimer);
   191→      userWs?.close();
   192→    };
   193→  }, []);
   194→
   195→  /* ── Click Outside ── */
   196→  useEffect(() => {
   197→    const handleClickOutside = (e: MouseEvent) => {
   198→      if (menuRef.current && !menuRef.current.contains(e.target as Node)) setUserMenuOpen(false);
   199→      if (notifRef.current && !notifRef.current.contains(e.target as Node)) setNotifOpen(false);
   200→    };
   201→    document.addEventListener("mousedown", handleClickOutside);
   202→    return () => document.removeEventListener("mousedown", handleClickOutside);
   203→  }, []);
   204→
   205→  /* ── Handlers ── */
   206→  const handleLogout = async () => {
   207→    const refreshToken = localStorage.getItem("refresh_token");
   208→    const token = localStorage.getItem("token");
   209→    try {
   210→      await fetch(`${process.env.NEXT_PUBLIC_API_URL || "http://localhost:3000"}/api/v1/auth/logout`, {
   211→        method: "POST",
   212→        headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` },
   213→        body: JSON.stringify({ refresh_token: refreshToken }),
   214→      });
   215→    } catch {}
   216→    localStorage.removeItem("token");
   217→    localStorage.removeItem("refresh_token");
   218→    localStorage.removeItem("user");
   219→    router.push("/login");
   220→  };
   221→
   222→  const handleMarkRead = async (id: number) => {
   223→    try {
   224→      await authPut(`/api/v1/notifications/${id}/read`);
   225→      setNotifications(prev => prev.map(n => n.id === id ? { ...n, read: true } : n));
   226→      setUnreadCount(prev => Math.max(0, prev - 1));
   227→    } catch {}
   228→  };
   229→
   230→  const handleMarkAllRead = async () => {
   231→    try {
   232→      await authPut("/api/v1/notifications/read-all");
   233→      setNotifications(prev => prev.map(n => ({ ...n, read: true })));
   234→      setUnreadCount(0);
   235→    } catch {}
   236→  };
   237→
   238→  const formatTimeAgo = (dateStr: string) => {
   239→    const diff = Date.now() - new Date(dateStr).getTime();
   240→    const mins = Math.floor(diff / 60000);
   241→    if (mins < 1) return "الآن";
   242→    if (mins < 60) return `منذ ${mins} دقيقة`;
   243→    const hours = Math.floor(mins / 60);
   244→    if (hours < 24) return `منذ ${hours} ساعة`;
   245→    const days = Math.floor(hours / 24);
   246→    return `منذ ${days} يوم`;
   247→  };
   248→
   249→  const formatPrice = (val: string | undefined) => {
   250→    if (!val) return "—";
   251→    const n = parseFloat(val);
   252→    if (n >= 1000) return n.toLocaleString("en-US", { minimumFractionDigits: 2, maximumFractionDigits: 2 });
   253→    return n.toLocaleString("en-US", { minimumFractionDigits: 2, maximumFractionDigits: 6 });
   254→  };
   255→
   256→  /* ── Loading ── */
   257→  if (!user) return (
   258→    <div className="min-h-screen bg-background flex items-center justify-center">
   259→      <div className="flex flex-col items-center gap-4">
   260→        <div className="spinner h-10 w-10" />
   261→        <span className="text-sm text-muted-foreground animate-pulse">جاري التحميل...</span>
   262→      </div>
   263→    </div>
   264→  );
   265→
   266→  /* ── Render ── */
   267→  return (
   268→    <div className="min-h-screen bg-background">
   269→      {/* ═══════════════ HEADER ═══════════════ */}
   270→      <header className="fixed top-0 left-0 right-0 z-40 h-16 border-b border-border/30"
   271→        style={{ background: "rgba(10, 12, 18, 0.75)", backdropFilter: "blur(24px) saturate(1.8)", WebkitBackdropFilter: "blur(24px) saturate(1.8)" }}
   272→      >
   273→        <div className="flex items-center justify-between h-full px-4 lg:px-6">
   274→          {/* Right side: Logo + Mobile toggle */}
   275→          <div className="flex items-center gap-3">
   276→            <button
   277→              onClick={() => setSidebarOpen(!sidebarOpen)}
   278→              className="btn-ghost p-2 md:hidden rounded-xl"
   279→              aria-label="فتح القائمة"
   280→            >
   281→              {sidebarOpen ? <X className="h-5 w-5" /> : <Menu className="h-5 w-5" />}
   282→            </button>
   283→            <Link href="/dashboard" className="flex items-center gap-2.5 group">
   284→              <div className="h-8 w-8 rounded-lg bg-primary/15 flex items-center justify-center glow-emerald">
   285→                <span className="text-primary font-black text-sm">E</span>
   286→              </div>
   287→              <span className="text-lg font-bold gradient-text hidden sm:block tracking-tight">EgMoney</span>
   288→            </Link>
   289→          </div>
   290→
   291→          {/* Center: Live Price Ticker */}
   292→          <div className="hidden md:flex items-center gap-1 lg:gap-5">
   293→            {TICKER_SYMBOLS.map(({ symbol, label, color }) => {
   294→              const pData = prices[symbol];
   295→              const price = pData?.price;
   296→              const change = pData?.change ?? 0;
   297→              const isUp = change >= 0;
   298→              return (
   299→                <div key={symbol} className="flex items-center gap-1.5 px-2.5 py-1 rounded-lg hover:bg-muted/20 transition-colors">
   300→                  <span className={`text-[11px] font-semibold ${color}`}>{label}</span>
   301→                  <span className="text-xs font-semibold text-foreground tabular-nums">
   302→                    ${formatPrice(price)}
   303→                  </span>
   304→                  <span className={`text-[10px] font-semibold tabular-nums flex items-center gap-0.5 ${isUp ? "text-emerald-400" : "text-red-400"}`}>
   305→                    {isUp ? <ArrowUpRight className="h-3 w-3" /> : <ArrowDownRight className="h-3 w-3" />}
   306→                    {Math.abs(change).toFixed(2)}%
   307→                  </span>
   308→                </div>
   309→              );
   310→            })}
   311→          </div>
   312→
   313→          {/* Left side: Notifications + User */}
   314→          <div className="flex items-center gap-2">
   315→            {/* Notification Bell */}
   316→            <div className="relative" ref={notifRef}>
   317→              <button
   318→                onClick={() => setNotifOpen(!notifOpen)}
   319→                className="btn-ghost p-2.5 rounded-xl relative group"
   320→                aria-label="الإشعارات"
   321→              >
   322→                <Bell className="h-5 w-5 transition-transform group-hover:rotate-12 duration-200" />
   323→                {unreadCount > 0 && (
   324→                  <span className="absolute top-1 right-1 h-4 min-w-[16px] bg-red-500 text-white text-[10px] font-bold rounded-full flex items-center justify-center px-1 animate-pulse shadow-[0_0_8px_rgba(239,68,68,0.5)]">
   325→                    {unreadCount > 99 ? "99+" : unreadCount}
   326→                  </span>
   327→                )}
   328→              </button>
   329→
   330→              {/* Notifications Dropdown */}
   331→              {notifOpen && (
   332→                <div className="absolute left-0 mt-3 w-80 glass-panel-strong rounded-2xl shadow-2xl border border-border/30 z-50 max-h-[420px] flex flex-col animate-slide-in-down overflow-hidden">
   333→                  {/* Header */}
   334→                  <div className="flex items-center justify-between p-4 border-b border-border/20">
   335→                    <div className="flex items-center gap-2">
   336→                      <h3 className="font-bold text-sm">الإشعارات</h3>
   337→                      {unreadCount > 0 && (
   338→                        <span className="text-[10px] font-semibold bg-primary/15 text-primary px-2 py-0.5 rounded-full">
   339→                          {unreadCount} جديد
   340→                        </span>
   341→                      )}
   342→                    </div>
   343→                    {unreadCount > 0 && (
   344→                      <button
   345→                        onClick={handleMarkAllRead}
   346→                        className="text-[11px] text-primary hover:text-primary/80 transition-colors flex items-center gap-1 font-medium"
   347→                      >
   348→                        <CheckCheck className="h-3.5 w-3.5" />
   349→                        تعيين الكل كمقروء
   350→                      </button>
   351→                    )}
   352→                  </div>
   353→
   354→                  {/* Notification List */}
   355→                  <div className="overflow-y-auto flex-1 max-h-80 custom-scrollbar">
   356→                    {notifications.length === 0 ? (
   357→                      <div className="p-8 text-center">
   358→                        <Bell className="h-8 w-8 text-muted-foreground/30 mx-auto mb-3" />
   359→                        <p className="text-sm text-muted-foreground">لا توجد إشعارات</p>
   360→                      </div>
   361→                    ) : (
   362→                      notifications.map((notif, i) => {
   363→                        const iconData = NOTIF_ICON_MAP[notif.type] || DEFAULT_NOTIF_ICON;
   364→                        const NotifIcon = iconData.icon;
   365→                        return (
   366→                          <div
   367→                            key={notif.id}
   368→                            className={`p-3.5 border-b border-border/10 cursor-pointer transition-all duration-200 hover:bg-muted/15 ${!notif.read ? "bg-primary/[0.04]" : ""}`}
   369→                            onClick={() => !notif.read && handleMarkRead(notif.id)}
   370→                            style={{ animationDelay: `${i * 40}ms` }}
   371→                          >
   372→                            <div className="flex items-start gap-3">
   373→                              {/* Icon */}
   374→                              <div className={`shrink-0 h-8 w-8 rounded-lg ${iconData.bg} flex items-center justify-center`}>
   375→                                <NotifIcon className={`h-4 w-4 ${iconData.fg}`} />
   376→                              </div>
   377→                              {/* Content */}
   378→                              <div className="flex-1 min-w-0">
   379→                                <div className="flex items-center gap-2">
   380→                                  <p className="text-sm font-semibold truncate">{notif.title}</p>
   381→                                  {!notif.read && (
   382→                                    <span className="shrink-0 h-2 w-2 rounded-full bg-primary shadow-[0_0_6px_rgba(16,185,129,0.5)] animate-pulse" />
   383→                                  )}
   384→                                </div>
   385→                                <p className="text-xs text-muted-foreground mt-1 line-clamp-2 leading-relaxed">{notif.body}</p>
   386→                                <p className="text-[10px] text-muted-foreground/60 mt-1.5">{formatTimeAgo(notif.created_at)}</p>
   387→                              </div>
   388→                            </div>
   389→                          </div>
   390→                        );
   391→                      })
   392→                    )}
   393→                  </div>
   394→
   395→                  {/* Footer */}
   396→                  {notifications.length > 0 && (
   397→                    <div className="p-3 border-t border-border/20">
   398→                      <Link
   399→                        href="/dashboard/notifications"
   400→                        onClick={() => setNotifOpen(false)}
   401→                        className="block text-center text-xs font-medium text-primary hover:text-primary/80 transition-colors py-1"
   402→                      >
   403→                        عرض جميع الإشعارات
   404→                      </Link>
   405→                    </div>
   406→                  )}
   407→                </div>
   408→              )}
   409→            </div>
   410→
   411→            {/* User Menu */}
   412→            <div className="relative" ref={menuRef}>
   413→              <button
   414→                onClick={() => setUserMenuOpen(!userMenuOpen)}
   415→                className="flex items-center gap-2.5 btn-ghost px-2 py-1.5 rounded-xl group"
   416→              >
   417→                {/* Avatar with gradient ring */}
   418→                <div className="relative">
   419→                  <div className="h-8 w-8 rounded-full bg-gradient-to-br from-emerald-400 via-teal-400 to-emerald-600 p-[2px]">
   420→                    <div className="h-full w-full rounded-full bg-card flex items-center justify-center">
   421→                      <span className="text-xs font-bold text-primary">{user.username?.charAt(0)?.toUpperCase() || "U"}</span>
   422→                    </div>
   423→                  </div>
   424→                  <span className="absolute -bottom-0.5 -left-0.5 h-2.5 w-2.5 rounded-full bg-emerald-500 border-2 border-card status-dot-online" />
   425→                </div>
   426→                <div className="hidden sm:flex flex-col items-start">
   427→                  <span className="text-sm font-semibold leading-tight">{user.username}</span>
   428→                  <span className="text-[10px] text-muted-foreground leading-tight">
   429→                    {user.role === "ADMIN" ? "مدير" : "مستخدم"}
   430→                  </span>
   431→                </div>
   432→                <ChevronDown className={`h-3.5 w-3.5 text-muted-foreground transition-transform duration-200 ${userMenuOpen ? "rotate-180" : ""}`} />
   433→              </button>
   434→
   435→              {/* User Dropdown */}
   436→              {userMenuOpen && (
   437→                <div className="absolute left-0 mt-3 w-56 glass-panel-strong rounded-2xl p-2 shadow-2xl border border-border/30 z-50 animate-slide-in-down overflow-hidden">
   438→                  {/* User Info */}
   439→                  <div className="px-3 py-3 mb-1">
   440→                    <div className="flex items-center gap-3">
   441→                      <div className="h-10 w-10 rounded-full bg-gradient-to-br from-emerald-400 via-teal-400 to-emerald-600 p-[2px]">
   442→                        <div className="h-full w-full rounded-full bg-card flex items-center justify-center">
   443→                          <span className="text-sm font-bold text-primary">{user.username?.charAt(0)?.toUpperCase() || "U"}</span>
   444→                        </div>
   445→                      </div>
   446→                      <div className="flex-1 min-w-0">
   447→                        <p className="text-sm font-semibold truncate">{user.username}</p>
   448→                        <p className="text-[11px] text-muted-foreground truncate">{user.email}</p>
   449→                      </div>
   450→                    </div>
   451→                    <div className="mt-2 flex items-center gap-1.5">
   452→                      <span className="text-[10px] font-medium bg-primary/10 text-primary px-2 py-0.5 rounded-full">
   453→                        {user.role === "ADMIN" ? "مدير النظام" : "مستخدم"}
   454→                      </span>
