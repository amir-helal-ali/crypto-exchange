    80→/* ─────────── Types ─────────── */
    81→interface Notification {
    82→  id: number;
    83→  type: string;
    84→  title: string;
    85→  body: string;
    86→  data: string;
    87→  read: boolean;
    88→  created_at: string;
    89→}
    90→
    91→/* ─────────── Icon Map (lucide-react) ─────────── */
    92→const NOTIF_ICON_MAP: Record<string, { icon: any; bg: string; fg: string }> = {
    93→  ORDER_FILLED: { icon: Check, bg: "bg-emerald-500/15", fg: "text-emerald-400" },
    94→  ORDER_CANCELLED: { icon: Ban, bg: "bg-red-500/15", fg: "text-red-400" },
    95→  ORDER_TRIGGERED: { icon: Zap, bg: "bg-amber-500/15", fg: "text-amber-400" },

    98→  WITHDRAWAL_REJECTED: { icon: AlertTriangle, bg: "bg-red-500/15", fg: "text-red-400" },
    99→  KYC_APPROVED: { icon: ShieldCheck, bg: "bg-emerald-500/15", fg: "text-emerald-400" },
   100→  KYC_REJECTED: { icon: AlertTriangle, bg: "bg-orange-500/15", fg: "text-orange-400" },
   101→};
   102→const DEFAULT_NOTIF_ICON = { icon: Bell, bg: "bg-primary/15", fg: "text-primary" };
   103→
   104→/* ─────────── Component ─────────── */
   105→export default function DashboardLayout({ children }: { children: React.ReactNode }) {
   106→  const router = useRouter();
   107→  const pathname = usePathname();
   108→  const [user, setUser] = useState<any>(null);
   109→  const [sidebarOpen, setSidebarOpen] = useState(false);
   110→  const [prices, setPrices] = useState<Record<string, { price: string; change: number }>>({});
   111→  const [prevPrices, setPrevPrices] = useState<Record<string, string>>({});
   112→  const [userMenuOpen, setUserMenuOpen] = useState(false);
   113→  const [notifOpen, setNotifOpen] = useState(false);
   114→  const [notifications, setNotifications] = useState<Notification[]>([]);
   115→  const [unreadCount, setUnreadCount] = useState(0);
   116→  const menuRef = useRef<HTMLDivElement>(null);
   117→  const notifRef = useRef<HTMLDivElement>(null);
   118→
   119→  /* ── Auth & Price WebSocket ── */
   120→  useEffect(() => {
   121→    const token = localStorage.getItem("token");
   122→    if (!token) { router.push("/login"); return; }
   123→    try { setUser(JSON.parse(localStorage.getItem("user") || "{}")); } catch { router.push("/login"); }
   124→
   125→    const ws = new WebSocket("wss://stream.binance.com:9443/ws/!miniTicker@arr");
   126→    ws.onmessage = (e) => {
   127→      const data = JSON.parse(e.data);
   128→      if (Array.isArray(data)) {
   129→        const updates: Record<string, { price: string; change: number }> = {};
   130→        const newPrev: Record<string, string> = {};
   131→        data.forEach((t: any) => {
   132→          if (["BTCUSDT", "ETHUSDT", "BNBUSDT", "SOLUSDT"].includes(t.s)) {
   133→            const prev = prevPrices[t.s];
   134→            const change = prev ? (parseFloat(t.c) - parseFloat(prev)) / parseFloat(prev) * 100 : 0;
   135→            updates[t.s] = { price: t.c, change };
   136→            newPrev[t.s] = t.c;
   137→          }
   138→        });
   139→        setPrices(prev => {
   140→          const merged = { ...prev };
   141→          Object.keys(updates).forEach(k => {
   142→            merged[k] = { ...prev[k], ...updates[k] };
   143→          });
   144→          return merged;
   145→        });
   146→        setPrevPrices(p => ({ ...p, ...newPrev }));
   147→      }
   148→    };
   149→    return () => ws.close();
   150→  }, [router]);
   151→
   152→  /* ── Notifications + User WebSocket ── */
   153→  useEffect(() => {
   154→    const fetchNotifs = async () => {
   155→      try {
   156→        const res = await authGet("/api/v1/notifications?limit=20");
   157→        if (res.ok) {
   158→          const data = await res.json();
   159→          setNotifications(data.data || []);
   160→          setUnreadCount(data.unread_count || 0);
   161→        }
   162→      } catch {}
   163→    };
   164→    fetchNotifs();
   165→    const interval = setInterval(fetchNotifs, 30000);
   166→
   167→    const token = localStorage.getItem("token");
   168→    const apiUrl = process.env.NEXT_PUBLIC_API_URL || "http://localhost:3000";
   169→    const wsUrl = apiUrl.replace(/^http/, "ws") + "/ws/user?token=" + token;
   170→    let userWs: WebSocket | null = null;
   171→    let reconnectTimer: number;
   172→
   173→    const connectUserWs = () => {
   174→      try {
   175→        userWs = new WebSocket(wsUrl);
   176→        userWs.onmessage = (e) => {
   177→          try {
   178→            const event = JSON.parse(e.data);
   179→            if (event.type === "order_fill") {
   180→              toast.success(`تم تنفيذ الأمر: ${event.data.side === "BUY" ? "شراء" : "بيع"} ${event.data.quantity} ${event.data.symbol}`);
   181→            } else if (event.type === "order_trigger") {
   182→              toast(`تم تفعيل الأمر الشرطي: ${event.data.symbol}`, { icon: "⚡" });
   183→            } else if (event.type === "deposit_approved") {
   184→              toast.success(`تم تأكيد الإيداع: ${event.data.amount} ${event.data.currency}`);
   185→            } else if (event.type === "withdrawal_approved") {
   186→              toast.success(`تم تأكيد السحب: ${event.data.amount} ${event.data.currency}`);
   187→            } else if (event.type === "withdrawal_rejected") {
   188→              toast.error(`تم رفض السحب: ${event.data.amount} ${event.data.currency}`);
   189→            } else if (event.type === "kyc_APPROVED") {
   190→              toast.success("تمت الموافقة على التحقق من الهوية");
   191→            } else if (event.type === "kyc_REJECTED") {
   192→              toast.error("تم رفض طلب التحقق من الهوية");
   193→            }
   194→            fetchNotifs();
   195→          } catch {}
   196→        };
   197→        userWs.onclose = () => { reconnectTimer = window.setTimeout(connectUserWs, 5000); };
   198→        userWs.onerror = () => { userWs?.close(); };
   199→      } catch {}
   200→    };
   201→    connectUserWs();
   202→
   203→    return () => {
   204→      clearInterval(interval);
   205→      clearTimeout(reconnectTimer);
   206→      userWs?.close();
   207→    };
   208→  }, []);
   209→
   210→  /* ── Click Outside ── */
   211→  useEffect(() => {
   212→    const handleClickOutside = (e: MouseEvent) => {
   213→      if (menuRef.current && !menuRef.current.contains(e.target as Node)) setUserMenuOpen(false);
   214→      if (notifRef.current && !notifRef.current.contains(e.target as Node)) setNotifOpen(false);
   215→    };
   216→    document.addEventListener("mousedown", handleClickOutside);
   217→    return () => document.removeEventListener("mousedown", handleClickOutside);
   218→  }, []);
   219→
   220→  /* ── Handlers ── */
   221→  const handleLogout = async () => {
   222→    const refreshToken = localStorage.getItem("refresh_token");
   223→    const token = localStorage.getItem("token");
   224→    try {
   225→      await fetch(`${process.env.NEXT_PUBLIC_API_URL || "http://localhost:3000"}/api/v1/auth/logout`, {
   226→        method: "POST",
   227→        headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` },
   228→        body: JSON.stringify({ refresh_token: refreshToken }),
   229→      });
   230→    } catch {}
   231→    localStorage.removeItem("token");
   232→    localStorage.removeItem("refresh_token");
   233→    localStorage.removeItem("user");
   234→    router.push("/login");
   235→  };
   236→
   237→  const handleMarkRead = async (id: number) => {
   238→    try {
   239→      await authPut(`/api/v1/notifications/${id}/read`);
   240→      setNotifications(prev => prev.map(n => n.id === id ? { ...n, read: true } : n));
   241→      setUnreadCount(prev => Math.max(0, prev - 1));
   242→    } catch {}
   243→  };
   244→
   245→  const handleMarkAllRead = async () => {
   246→    try {
   247→      await authPut("/api/v1/notifications/read-all");
   248→      setNotifications(prev => prev.map(n => ({ ...n, read: true })));
   249→      setUnreadCount(0);
   250→    } catch {}
   251→  };
   252→
   253→  const formatTimeAgo = (dateStr: string) => {
   254→    const diff = Date.now() - new Date(dateStr).getTime();
   255→    const mins = Math.floor(diff / 60000);
   256→    if (mins < 1) return "الآن";
   257→    if (mins < 60) return `منذ ${mins} دقيقة`;
   258→    const hours = Math.floor(mins / 60);
   259→    if (hours < 24) return `منذ ${hours} ساعة`;
   260→    const days = Math.floor(hours / 24);
   261→    return `منذ ${days} يوم`;
   262→  };
   263→
   264→  const formatPrice = (val: string | undefined) => {
   265→    if (!val) return "—";
   266→    const n = parseFloat(val);
   267→    if (n >= 1000) return n.toLocaleString("en-US", { minimumFractionDigits: 2, maximumFractionDigits: 2 });
   268→    return n.toLocaleString("en-US", { minimumFractionDigits: 2, maximumFractionDigits: 6 });
   269→  };
   270→
   271→  /* ── Loading ── */
   272→  if (!user) return (
   273→    <div className="min-h-screen bg-background flex items-center justify-center">
   274→      <div className="flex flex-col items-center gap-4">
   275→        <div className="spinner h-10 w-10" />
   276→        <span className="text-sm text-muted-foreground animate-pulse">جاري التحميل...</span>
   277→      </div>
   278→    </div>
   279→  );
   280→
   281→  /* ── Render ── */
   282→  return (
   283→    <div className="min-h-screen bg-background">
   284→      {/* ═══════════════ HEADER ═══════════════ */}
   285→      <header className="fixed top-0 left-0 right-0 z-40 h-16 border-b border-border/30"
   286→        style={{ background: "rgba(10, 12, 18, 0.75)", backdropFilter: "blur(24px) saturate(1.8)", WebkitBackdropFilter: "blur(24px) saturate(1.8)" }}
   287→      >
   288→        <div className="flex items-center justify-between h-full px-4 lg:px-6">
   289→          {/* Right side: Logo + Mobile toggle */}
   290→          <div className="flex items-center gap-3">
   291→            <button
   292→              onClick={() => setSidebarOpen(!sidebarOpen)}
   293→              className="btn-ghost p-2 md:hidden rounded-xl"
   294→              aria-label="فتح القائمة"
   295→            >
   296→              {sidebarOpen ? <X className="h-5 w-5" /> : <Menu className="h-5 w-5" />}
   297→            </button>
   298→            <Link href="/dashboard" className="flex items-center gap-2.5 group">
   299→              <div className="h-8 w-8 rounded-lg bg-primary/15 flex items-center justify-center glow-emerald">
   300→                <span className="text-primary font-black text-sm">E</span>
   301→              </div>
   302→              <span className="text-lg font-bold gradient-text hidden sm:block tracking-tight">EgMoney</span>
   303→            </Link>
   304→          </div>
   305→
   306→          {/* Center: Live Price Ticker */}
   307→          <div className="hidden md:flex items-center gap-1 lg:gap-5">
   308→            {TICKER_SYMBOLS.map(({ symbol, label, color }) => {
   309→              const pData = prices[symbol];
   310→              const price = pData?.price;
   311→              const change = pData?.change ?? 0;
   312→              const isUp = change >= 0;
   313→              return (
   314→                <div key={symbol} className="flex items-center gap-1.5 px-2.5 py-1 rounded-lg hover:bg-muted/20 transition-colors">
   315→                  <span className={`text-[11px] font-semibold ${color}`}>{label}</span>
   316→                  <span className="text-xs font-semibold text-foreground tabular-nums">
   317→                    ${formatPrice(price)}
   318→                  </span>
   319→                  <span className={`text-[10px] font-semibold tabular-nums flex items-center gap-0.5 ${isUp ? "text-emerald-400" : "text-red-400"}`}>
   320→                    {isUp ? <ArrowUpRight className="h-3 w-3" /> : <ArrowDownRight className="h-3 w-3" />}
   321→                    {Math.abs(change).toFixed(2)}%
   322→                  </span>
   323→                </div>
   324→              );
   325→            })}
   326→          </div>
   327→
   328→          {/* Left side: Notifications + User */}
   329→          <div className="flex items-center gap-2">
   330→            {/* Notification Bell */}
   331→            <div className="relative" ref={notifRef}>
   332→              <button
   333→                onClick={() => setNotifOpen(!notifOpen)}
   334→                className="btn-ghost p-2.5 rounded-xl relative group"
   335→                aria-label="الإشعارات"
   336→              >
   337→                <Bell className="h-5 w-5 transition-transform group-hover:rotate-12 duration-200" />
   338→                {unreadCount > 0 && (
   339→                  <span className="absolute top-1 right-1 h-4 min-w-[16px] bg-red-500 text-white text-[10px] font-bold rounded-full flex items-center justify-center px-1 animate-pulse shadow-[0_0_8px_rgba(239,68,68,0.5)]">
   340→                    {unreadCount > 99 ? "99+" : unreadCount}
   341→                  </span>
   342→                )}
   343→              </button>
   344→
   345→              {/* Notifications Dropdown */}
   346→              {notifOpen && (
   347→                <div className="absolute left-0 mt-3 w-80 glass-panel-strong rounded-2xl shadow-2xl border border-border/30 z-50 max-h-[420px] flex flex-col animate-slide-in-down overflow-hidden">
   348→                  {/* Header */}
   349→                  <div className="flex items-center justify-between p-4 border-b border-border/20">
   350→                    <div className="flex items-center gap-2">
   351→                      <h3 className="font-bold text-sm">الإشعارات</h3>
   352→                      {unreadCount > 0 && (
   353→                        <span className="text-[10px] font-semibold bg-primary/15 text-primary px-2 py-0.5 rounded-full">
   354→                          {unreadCount} جديد
   355→                        </span>
   356→                      )}
   357→                    </div>
   358→                    {unreadCount > 0 && (
   359→                      <button
   360→                        onClick={handleMarkAllRead}
   361→                        className="text-[11px] text-primary hover:text-primary/80 transition-colors flex items-center gap-1 font-medium"
   362→                      >
   363→                        <CheckCheck className="h-3.5 w-3.5" />
   364→                        تعيين الكل كمقروء
   365→                      </button>
   366→                    )}
   367→                  </div>
   368→
   369→                  {/* Notification List */}
   370→                  <div className="overflow-y-auto flex-1 max-h-80 custom-scrollbar">
   371→                    {notifications.length === 0 ? (
   372→                      <div className="p-8 text-center">
   373→                        <Bell className="h-8 w-8 text-muted-foreground/30 mx-auto mb-3" />
   374→                        <p className="text-sm text-muted-foreground">لا توجد إشعارات</p>
   375→                      </div>
   376→                    ) : (
   377→                      notifications.map((notif, i) => {
   378→                        const iconData = NOTIF_ICON_MAP[notif.type] || DEFAULT_NOTIF_ICON;
   379→                        const NotifIcon = iconData.icon;
   380→                        return (
   381→                          <div
   382→                            key={notif.id}
   383→                            className={`p-3.5 border-b border-border/10 cursor-pointer transition-all duration-200 hover:bg-muted/15 ${!notif.read ? "bg-primary/[0.04]" : ""}`}
   384→                            onClick={() => !notif.read && handleMarkRead(notif.id)}
   385→                            style={{ animationDelay: `${i * 40}ms` }}
   386→                          >
   387→                            <div className="flex items-start gap-3">
   388→                              {/* Icon */}
   389→                              <div className={`shrink-0 h-8 w-8 rounded-lg ${iconData.bg} flex items-center justify-center`}>
   390→                                <NotifIcon className={`h-4 w-4 ${iconData.fg}`} />
   391→                              </div>
   392→                              {/* Content */}
   393→                              <div className="flex-1 min-w-0">
   394→                                <div className="flex items-center gap-2">
   395→                                  <p className="text-sm font-semibold truncate">{notif.title}</p>
   396→                                  {!notif.read && (
   397→                                    <span className="shrink-0 h-2 w-2 rounded-full bg-primary shadow-[0_0_6px_rgba(16,185,129,0.5)] animate-pulse" />
   398→                                  )}
   399→                                </div>
   400→                                <p className="text-xs text-muted-foreground mt-1 line-clamp-2 leading-relaxed">{notif.body}</p>
   401→                                <p className="text-[10px] text-muted-foreground/60 mt-1.5">{formatTimeAgo(notif.created_at)}</p>
   402→                              </div>
   403→                            </div>
   404→                          </div>
   405→                        );
   406→                      })
   407→                    )}
   408→                  </div>
   409→
   410→                  {/* Footer */}
   411→                  {notifications.length > 0 && (
   412→                    <div className="p-3 border-t border-border/20">
   413→                      <Link
   414→                        href="/dashboard/notifications"
   415→                        onClick={() => setNotifOpen(false)}
   416→                        className="block text-center text-xs font-medium text-primary hover:text-primary/80 transition-colors py-1"
   417→                      >
   418→                        عرض جميع الإشعارات
   419→                      </Link>
   420→                    </div>
   421→                  )}
   422→                </div>
   423→              )}
   424→            </div>
   425→
   426→            {/* User Menu */}
   427→            <div className="relative" ref={menuRef}>
   428→              <button
   429→                onClick={() => setUserMenuOpen(!userMenuOpen)}
   430→                className="flex items-center gap-2.5 btn-ghost px-2 py-1.5 rounded-xl group"
   431→              >
   432→                {/* Avatar with gradient ring */}
   433→                <div className="relative">
   434→                  <div className="h-8 w-8 rounded-full bg-gradient-to-br from-emerald-400 via-teal-400 to-emerald-600 p-[2px]">
   435→                    <div className="h-full w-full rounded-full bg-card flex items-center justify-center">
   436→                      <span className="text-xs font-bold text-primary">{user.username?.charAt(0)?.toUpperCase() || "U"}</span>
   437→                    </div>
   438→                  </div>
   439→                  <span className="absolute -bottom-0.5 -left-0.5 h-2.5 w-2.5 rounded-full bg-emerald-500 border-2 border-card status-dot-online" />
   440→                </div>
   441→                <div className="hidden sm:flex flex-col items-start">
   442→                  <span className="text-sm font-semibold leading-tight">{user.username}</span>
   443→                  <span className="text-[10px] text-muted-foreground leading-tight">
   444→                    {user.role === "ADMIN" ? "مدير" : "مستخدم"}
   445→                  </span>
   446→                </div>
   447→                <ChevronDown className={`h-3.5 w-3.5 text-muted-foreground transition-transform duration-200 ${userMenuOpen ? "rotate-180" : ""}`} />
   448→              </button>
   449→
   450→              {/* User Dropdown */}
   451→              {userMenuOpen && (
   452→                <div className="absolute left-0 mt-3 w-56 glass-panel-strong rounded-2xl p-2 shadow-2xl border border-border/30 z-50 animate-slide-in-down overflow-hidden">
   453→                  {/* User Info */}
   454→                  <div className="px-3 py-3 mb-1">
   455→                    <div className="flex items-center gap-3">
