    78→/* ─── Intersection Observer Hook ─── */
    79→function useInView(threshold = 0.15) {
    80→  const ref = useRef<HTMLDivElement>(null);
    81→  const [isInView, setIsInView] = useState(false);
    82→
    83→  useEffect(() => {
    84→    const el = ref.current;
    85→    if (!el) return;
    86→    const obs = new IntersectionObserver(
    87→      ([entry]) => { if (entry.isIntersecting) { setIsInView(true); obs.unobserve(el); } },
    88→      { threshold }
    89→    );
    90→    obs.observe(el);
    91→    return () => obs.disconnect();
    92→  }, [threshold]);
    93→
    94→  return { ref, isInView };
    95→}

    97→/* ─── Count-Up Hook ─── */
    98→function useCountUp(target: number, duration: number, start: boolean) {
    99→  const [value, setValue] = useState(0);
   100→  useEffect(() => {
   101→    if (!start) return;
   102→    let startTime: number | null = null;
   103→    let raf: number;
   104→    const animate = (ts: number) => {
   105→      if (!startTime) startTime = ts;
   106→      const progress = Math.min((ts - startTime) / duration, 1);
   107→      const eased = 1 - Math.pow(1 - progress, 3);
   108→      setValue(eased * target);
   109→      if (progress < 1) raf = requestAnimationFrame(animate);
   110→    };
   111→    raf = requestAnimationFrame(animate);
   112→    return () => cancelAnimationFrame(raf);
   113→  }, [target, duration, start]);
   114→  return value;
   115→}

   117→/* ─── Animated Particles Component ─── */
   118→function ParticleField() {
   119→  const canvasRef = useRef<HTMLCanvasElement>(null);
   120→
   121→  useEffect(() => {
   122→    const canvas = canvasRef.current;
   123→    if (!canvas) return;
   124→    const ctx = canvas.getContext("2d");
   125→    if (!ctx) return;
   126→
   127→    let animId: number;
   128→    const particles: { x: number; y: number; vx: number; vy: number; r: number; opacity: number; color: string }[] = [];
   129→    const colors = ["#10b981", "#14b8a6", "#627eea", "#9945ff", "#f7931a"];
   130→
   131→    const resize = () => {
   132→      canvas.width = window.innerWidth;
   133→      canvas.height = window.innerHeight;
   134→    };
   135→    resize();
   136→    window.addEventListener("resize", resize);
   137→
   138→    for (let i = 0; i < 60; i++) {
   139→      particles.push({
   140→        x: Math.random() * canvas.width,
   141→        y: Math.random() * canvas.height,
   142→        vx: (Math.random() - 0.5) * 0.4,
   143→        vy: (Math.random() - 0.5) * 0.4,
   144→        r: Math.random() * 2 + 1,
   145→        opacity: Math.random() * 0.4 + 0.1,
   146→        color: colors[Math.floor(Math.random() * colors.length)],
   147→      });
   148→    }
   149→
   150→    const draw = () => {
   151→      ctx.clearRect(0, 0, canvas.width, canvas.height);
   152→      particles.forEach((p) => {
   153→        p.x += p.vx;
   154→        p.y += p.vy;
   155→        if (p.x < 0) p.x = canvas.width;
   156→        if (p.x > canvas.width) p.x = 0;
   157→        if (p.y < 0) p.y = canvas.height;
   158→        if (p.y > canvas.height) p.y = 0;
   159→
   160→        ctx.beginPath();
   161→        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
   162→        ctx.fillStyle = p.color;
   163→        ctx.globalAlpha = p.opacity;
   164→        ctx.fill();
   165→      });
   166→
   167→      // Draw lines between close particles
   168→      ctx.globalAlpha = 0.06;
   169→      ctx.strokeStyle = "#10b981";
   170→      ctx.lineWidth = 0.5;
   171→      for (let i = 0; i < particles.length; i++) {
   172→        for (let j = i + 1; j < particles.length; j++) {
   173→          const dx = particles[i].x - particles[j].x;
   174→          const dy = particles[i].y - particles[j].y;
   175→          const dist = Math.sqrt(dx * dx + dy * dy);
   176→          if (dist < 150) {
   177→            ctx.globalAlpha = 0.06 * (1 - dist / 150);
   178→            ctx.beginPath();
   179→            ctx.moveTo(particles[i].x, particles[i].y);
   180→            ctx.lineTo(particles[j].x, particles[j].y);
   181→            ctx.stroke();
   182→          }
   183→        }
   184→      }
   185→      ctx.globalAlpha = 1;
   186→      animId = requestAnimationFrame(draw);
   187→    };
   188→    draw();
   189→
   190→    return () => {
   191→      cancelAnimationFrame(animId);
   192→      window.removeEventListener("resize", resize);
   193→    };
   194→  }, []);
   195→
   196→  return <canvas ref={canvasRef} className="absolute inset-0 w-full h-full pointer-events-none" />;
   197→}

   346→/* ─── Price Card Component ─── */
   347→function PriceCard({
   348→  coin,
   349→  priceData,
   350→  canvasRef,
   351→  drawSparkline,
   352→}: {
   353→  coin: typeof COINS[0];
   354→  priceData: { price: number; high: number; low: number; volume: string; change: string } | undefined;
   355→  canvasRef: (el: HTMLCanvasElement | null) => void;
   356→  drawSparkline: (symbol: string) => void;
   357→}) {
   358→  const [flash, setFlash] = useState<"up" | "down" | null>(null);
   359→  const prevPriceRef = useRef<number | null>(null);
   360→
   361→  useEffect(() => {
   362→    if (!priceData) return;
   363→    if (prevPriceRef.current !== null && prevPriceRef.current !== priceData.price) {
   364→      setFlash(priceData.price > prevPriceRef.current ? "up" : "down");
   365→      const t = setTimeout(() => setFlash(null), 600);
   366→      return () => clearTimeout(t);
   367→    }
   368→    prevPriceRef.current = priceData.price;
   369→  }, [priceData?.price]);
   370→
   371→  const changeVal = priceData?.change ? parseFloat(priceData.change) : 0;
   372→  const isPositive = changeVal >= 0;
   373→
   374→  return (
   375→    <div className="glass-panel rounded-2xl p-4 sm:p-5 card-hover group relative overflow-hidden">
   376→      {/* Glow accent on hover */}
   377→      <div
   378→        className="absolute top-0 left-0 w-full h-1 opacity-0 group-hover:opacity-100 transition-opacity duration-300"
   379→        style={{ background: `linear-gradient(90deg, ${coin.color}80, transparent)` }}
   380→      />
   381→
   382→      <div className="flex items-center justify-between mb-3">
   383→        <div className="flex items-center gap-3">
   384→          <div
   385→            className="h-10 w-10 rounded-full flex items-center justify-center text-sm font-bold shrink-0 ring-2 ring-offset-2 ring-offset-background"
   386→            style={{ backgroundColor: coin.color + "20", color: coin.color, "--tw-ring-color": coin.color + "40" } as React.CSSProperties}
   387→          >
   388→            {coin.name.charAt(0)}
   389→          </div>
   390→          <div>
   391→            <p className="font-bold text-sm">{coin.arName}</p>
   392→            <p className="text-[11px] text-muted-foreground">{coin.name}</p>
   393→          </div>
   394→        </div>
   395→        <div
   396→          className={`px-2 py-1 rounded-lg text-[11px] font-bold ${
   397→            isPositive ? "bg-emerald-500/10 text-emerald-500" : "bg-red-500/10 text-red-500"
   398→          }`}
   399→        >
   400→          {priceData?.change ? `${isPositive ? "+" : ""}${priceData.change}%` : "—"}
   401→        </div>
   402→      </div>
   403→
   404→      <div className="flex items-end justify-between">
   405→        <div>
   406→          <p
   407→            className={`text-xl sm:text-2xl font-bold tabular-nums transition-colors duration-300 ${
   408→              flash === "up" ? "text-emerald-500" : flash === "down" ? "text-red-500" : ""
   409→            }`}
   410→          >
   411→            {priceData?.price?.toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 }) || "—"}
   412→          </p>
   413→          <p className="text-[10px] text-muted-foreground mt-1">
   414→            حجم: {priceData?.volume ? (parseFloat(priceData.volume) / 1000).toFixed(0) + "K" : "—"}
   415→          </p>
   416→        </div>
   417→        <canvas
   418→          ref={(el) => { canvasRef(el); if (el) drawSparkline(coin.symbol); }}
   419→          className="w-24 h-12 rounded-lg"
   420→        />
   421→      </div>
   422→    </div>
   423→  );
   424→}
