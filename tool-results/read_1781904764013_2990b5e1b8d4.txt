    47→  /* ─────── State ─────── */

    49→  const [prices, setPrices] = useState<Record<string, TickerData>>({});
    50→  const [orderBook, setOrderBook] = useState<{
    51→    bids: [string, string][];
    52→    asks: [string, string][];
    53→  }>({ bids: [], asks: [] });
    54→  const [side, setSide] = useState<OrderSide>("BUY");
    55→  const [orderType, setOrderType] = useState<OrderType>("MARKET");
    56→  const [orders, setOrders] = useState<UserOrder[]>([]);
    57→  const [wallets, setWallets] = useState<WalletType[]>([]);
    58→  const [feeSchedules, setFeeSchedules] = useState<FeeSchedule[]>([]);
    59→  const [loading, setLoading] = useState(false);
    60→  const [timeframe, setTimeframe] = useState<Timeframe>("15m");
    61→  const [candles, setCandles] = useState<Candle[]>([]);
    62→  const [marketTrades, setMarketTrades] = useState<MarketTrade[]>([]);
    63→  const [crosshairData, setCrosshairData] = useState<Candle | null>(null);
    64→  const [chartType, setChartType] = useState<"candles" | "line" | "area">(
    65→    "candles"
    66→  );
    67→  const [marketListOpen, setMarketListOpen] = useState(true);
    68→  const [chartFullscreen, setChartFullscreen] = useState(false);
    69→  const [soundEnabled, setSoundEnabled] = useState(true);
    70→  const [indicators, setIndicators] = useState<ChartIndicators>({
    71→    sma20: true,
    72→    ema50: true,
    73→    bollinger: false,
    74→    rsi: false,
    75→    macd: false,
    76→    vwap: false,
    77→    stochastic: false,
    78→    ichimoku: false,
    79→    volume: true,
    80→  });
    81→  const [showDepthChart, setShowDepthChart] = useState(true);
    82→
    83→  /* Drawing tools state */
    84→  const [activeTool, setActiveTool] = useState<DrawingTool>("cursor");
    85→  const [drawings, setDrawings] = useState<Drawing[]>([]);
    86→  const [drawingColor, setDrawingColor] = useState<string>(DRAWING_COLORS[0]);
    87→
    88→  /* Quick pair search state */
    89→  const [searchOpen, setSearchOpen] = useState(false);
    90→  const [favorites, setFavorites] = useState<string[]>([]);
    91→
    92→  /* Mobile active tab state */
    93→  const [mobileTab, setMobileTab] = useState<MobileTab>("chart");
    94→
    95→  const chartRef = useRef<ProChartHandle>(null);
    96→  const wsRef = useRef<WebSocket | null>(null);
    97→  const prevPrices = useRef<Record<string, number>>({});
    98→
    99→  /* ─────── Data Fetching ─────── */
   100→  const fetchOrders = useCallback(async () => {
   101→    try {
   102→      const r = await authGet("/api/v1/exchange/orders");
   103→      const d = await r.json();
   104→      setOrders(Array.isArray(d.data) ? d.data : []);
   105→    } catch {}
   106→  }, []);
   107→
   108→  const fetchWallets = useCallback(async () => {
   109→    try {
   110→      const r = await authGet("/api/v1/wallet/balances");
   111→      const d = await r.json();
   112→      setWallets(Array.isArray(d.data) ? d.data : []);
   113→    } catch {}
   114→  }, []);
   115→
   116→  const fetchFees = useCallback(async () => {
   117→    try {
   118→      const r = await fetch(
   119→        `${process.env.NEXT_PUBLIC_API_URL || "http://localhost:3000"}/api/v1/fees`
   120→      );
   121→      if (r.ok) {
   122→        const d = await r.json();
   123→        setFeeSchedules(Array.isArray(d.data) ? d.data : []);
   124→      }
   125→    } catch {}
   126→  }, []);
   127→
   128→  const fetchKlines = useCallback(async () => {
   129→    try {
   130→      const res = await fetch(
   131→        `https://api.binance.com/api/v3/klines?symbol=${selectedPair}&interval=${timeframe}&limit=500`
   132→      );
   133→      if (!res.ok) return;
   134→      const data = await res.json();
   135→      const candleData: Candle[] = data.map((k: any) => ({
   136→        time: Math.floor(k[0] / 1000),
   137→        open: parseFloat(k[1]),
   138→        high: parseFloat(k[2]),
   139→        low: parseFloat(k[3]),
   140→        close: parseFloat(k[4]),
   141→        volume: parseFloat(k[5]),
   142→      }));
   143→      setCandles(candleData);
   144→    } catch (e) {
   145→      console.error("Failed to fetch klines:", e);
   146→    }
   147→  }, [selectedPair, timeframe]);
   148→
   149→  const fetchMarketTrades = useCallback(async () => {
   150→    try {
   151→      const res = await fetch(
   152→        `https://api.binance.com/api/v3/trades?symbol=${selectedPair}&limit=30`
   153→      );
   154→      if (!res.ok) return;
   155→      const data = await res.json();
   156→      setMarketTrades(
   157→        data.map((t: any) => ({
   158→          id: t.id,
   159→          price: parseFloat(t.price),
   160→          qty: parseFloat(t.qty),
   161→          time: t.time,
   162→          isBuyerMaker: t.isBuyerMaker,
   163→        }))
   164→      );
   165→    } catch {}
   166→  }, [selectedPair]);
   167→
   168→  /* Initial fetch */
   169→  useEffect(() => {
   170→    fetchOrders();
   171→    fetchWallets();
   172→    fetchFees();
   173→  }, [fetchOrders, fetchWallets, fetchFees]);
   174→
   175→  /* Load drawings for the selected pair from localStorage */
   176→  useEffect(() => {
   177→    try {
   178→      const raw = localStorage.getItem(`exchange_drawings_${selectedPair}`);
   179→      if (raw) {
   180→        setDrawings(JSON.parse(raw));
   181→      } else {
   182→        setDrawings([]);
   183→      }
   184→    } catch {
   185→      setDrawings([]);
   186→    }
   187→  }, [selectedPair]);
   188→
   189→  /* Persist drawings to localStorage whenever they change */
   190→  useEffect(() => {
   191→    try {
   192→      localStorage.setItem(
   193→        `exchange_drawings_${selectedPair}`,
   194→        JSON.stringify(drawings)
   195→      );
   196→    } catch {}
   197→  }, [drawings, selectedPair]);
   198→
   199→  /* Refetch when pair/timeframe changes */
   200→  useEffect(() => {
   201→    fetchKlines();
   202→    fetchMarketTrades();
   203→  }, [fetchKlines, fetchMarketTrades]);
   204→
   205→  /* ─────── Binance WebSocket (Ticker + Depth + All tickers tape) ─────── */
   206→  useEffect(() => {
   207→    const ws = new WebSocket("wss://stream.binance.com:9443/ws");
   208→    const streams = [
   209→      `${selectedPair.toLowerCase()}@ticker`,
   210→      `${selectedPair.toLowerCase()}@depth20@100ms`,
   211→      "!ticker@arr", // all tickers for the tape
   212→    ];
   213→    ws.onopen = () =>
   214→      ws.send(JSON.stringify({ method: "SUBSCRIBE", params: streams, id: 1 }));
   215→
   216→    ws.onmessage = (e) => {
   217→      const d = JSON.parse(e.data);
   218→
   219→      // Single ticker update
   220→      if (d.e === "24hrTicker" && d.s === selectedPair) {
   221→        prevPrices.current[selectedPair] =
   222→          prices[selectedPair]?.price || parseFloat(d.c);
   223→        setPrices((prev: any) => ({
   224→          ...prev,
   225→          [selectedPair]: {
   226→            price: parseFloat(d.c),
   227→            high: parseFloat(d.h),
   228→            low: parseFloat(d.l),
   229→            volume: d.v,
   230→            change: d.P,
   231→            quoteVolume: d.q,
   232→            openPrice: d.o,
   233→          },
   234→        }));
   235→      }
   236→      // All tickers array (for ticker tape)
   237→      else if (Array.isArray(d) && d.length > 0 && d[0]?.e === "24hrTicker") {
   238→        setPrices((prev: any) => {
   239→          const next = { ...prev };
   240→          for (const t of d) {
   241→            if (t.s && t.s.endsWith("USDT")) {
   242→              next[t.s] = {
   243→                price: parseFloat(t.c),
   244→                high: parseFloat(t.h),
   245→                low: parseFloat(t.l),
   246→                volume: t.v,
   247→                change: t.P,
   248→                quoteVolume: t.q,
   249→                openPrice: t.o,
   250→              };
   251→            }
   252→          }
   253→          return next;
   254→        });
   255→      }
   256→      // Depth update
   257→      else if (d.e === "depthUpdate" && d.bids && d.asks) {
   258→        setOrderBook({
   259→          bids: d.bids.slice(0, 15),
   260→          asks: d.asks.slice(0, 15),
   261→        });
   262→      }
   263→    };
   264→
   265→    wsRef.current = ws;
   266→    return () => ws.close();
   267→    // eslint-disable-next-line react-hooks/exhaustive-deps
   268→  }, [selectedPair]);
   269→
   270→  /* ─────── Real-time Kline WebSocket ─────── */
   271→  useEffect(() => {
   272→    const ws = new WebSocket("wss://stream.binance.com:9443/ws");
   273→    ws.onopen = () =>
   274→      ws.send(
   275→        JSON.stringify({
   276→          method: "SUBSCRIBE",
   277→          params: [`${selectedPair.toLowerCase()}@kline_${timeframe}`],
   278→          id: 1,
   279→        })
   280→      );
   281→    ws.onmessage = (e) => {
   282→      const d = JSON.parse(e.data);
   283→      if (d.e === "kline" && chartRef.current) {
   284→        const k = d.k;
   285→        chartRef.current.updateLastCandle({
   286→          time: Math.floor(k.t / 1000),
   287→          open: parseFloat(k.o),
   288→          high: parseFloat(k.h),
   289→          low: parseFloat(k.l),
   290→          close: parseFloat(k.c),
   291→          volume: parseFloat(k.v),
   292→        });
   293→      }
   294→    };
   295→    return () => ws.close();
   296→  }, [selectedPair, timeframe]);
   297→
   298→  /* ─────── Real-time Trade WebSocket ─────── */
   299→  useEffect(() => {
   300→    const ws = new WebSocket("wss://stream.binance.com:9443/ws");
   301→    ws.onopen = () =>
   302→      ws.send(
   303→        JSON.stringify({
   304→          method: "SUBSCRIBE",
   305→          params: [`${selectedPair.toLowerCase()}@trade`],
   306→          id: 1,
   307→        })
   308→      );
   309→    ws.onmessage = (e) => {
   310→      const d = JSON.parse(e.data);
   311→      if (d.e === "trade") {
   312→        setMarketTrades((prev) => {
   313→          const newTrade: MarketTrade = {
   314→            id: d.t,
   315→            price: parseFloat(d.p),
   316→            qty: parseFloat(d.q),
   317→            time: d.T,
   318→            isBuyerMaker: d.m,
   319→          };
   320→          // dedupe
   321→          if (prev.some((t) => t.id === newTrade.id)) return prev;
   322→          return [newTrade, ...prev].slice(0, 50);
   323→        });
   324→      }
   325→    };
   326→    return () => ws.close();
   327→  }, [selectedPair]);
   328→
   329→  /* ─────── Order Handlers ─────── */
   330→  const handleSubmit = async (form: {
   331→    quantity: string;
   332→    price: string;
   333→    stopPrice: string;
   334→  }) => {
   335→    setLoading(true);
   336→    try {
   337→      const body: any = {
   338→        symbol: selectedPair,
   339→        side,
   340→        type: orderType,
   341→        quantity: form.quantity,
   342→      };
   343→      if (orderType === "LIMIT" || orderType === "STOP_LIMIT")
   344→        body.price = form.price;
   345→      if (orderType === "STOP_LIMIT" || orderType === "TAKE_PROFIT")
   346→        body.stop_price = form.stopPrice;
   347→
   348→      const res = await authPost("/api/v1/exchange/order", body);
   349→      const data = await res.json();
   350→      if (!res.ok) {
   351→        toast.error(data.error || "فشل تنفيذ الأمر");
   352→        getSoundManager().play("error");
   353→        return;
   354→      }
   355→      toast.success(data.message || "تم تنفيذ الأمر بنجاح");
   356→      // Different sound: filled vs pending
   357→      if (data.data?.status === "FILLED" || data.data?.status === "filled") {
   358→        getSoundManager().play("order_filled");
   359→      } else {
   360→        getSoundManager().play("order_placed");
   361→      }
   362→      fetchOrders();
   363→      fetchWallets();
   364→    } catch {
   365→      toast.error("حدث خطأ في الاتصال");
   366→      getSoundManager().play("error");
   367→    } finally {
   368→      setLoading(false);
   369→    }
   370→  };
   371→
   372→  const handleCancel = async (orderId: number) => {
   373→    try {
   374→      const res = await authPost(`/api/v1/exchange/order/${orderId}/cancel`);
   375→      const data = await res.json();
   376→      if (!res.ok) {
   377→        toast.error(data.error || "فشل إلغاء الأمر");
   378→        getSoundManager().play("error");
   379→        return;
   380→      }
   381→      toast.success(data.message || "تم إلغاء الأمر بنجاح");
   382→      getSoundManager().play("order_cancelled");
   383→      fetchOrders();
   384→      fetchWallets();
   385→    } catch {
   386→      toast.error("حدث خطأ في الاتصال");
   387→    }
   388→  };
   389→
   390→  /* ─────── Keyboard shortcuts ─────── */
   391→  useKeyboardShortcuts({
   392→    onToggleSide: () => setSide((s) => (s === "BUY" ? "SELL" : "BUY")),
   393→    onSetBuy: () => setSide("BUY"),
   394→    onSetSell: () => setSide("SELL"),
   395→    onSetMarket: () => setOrderType("MARKET"),
   396→    onSetLimit: () => setOrderType("LIMIT"),
   397→    onOpenSearch: () => setSearchOpen(true),
   398→    onExportChart: () => {
   399→      if (chartRef.current) {
   400→        chartRef.current.exportPng(`chart_${selectedPair}_${timeframe}.png`);
   401→        toast.success("تم تصدير صورة الشارت");
   402→      }
   403→    },
   404→    onToggleFullscreen: () => setChartFullscreen((v) => !v),
   405→  });
   406→
   407→  /* ─────── Load favorites from localStorage ─────── */
   408→  useEffect(() => {
   409→    try {
   410→      const raw = localStorage.getItem("exchange_favorites");
   411→      if (raw) setFavorites(JSON.parse(raw));
   412→    } catch {}
   413→    // Listen for changes from MarketList or QuickPairSearch
   414→    const syncFavorites = () => {
   415→      try {
   416→        const stored = localStorage.getItem("exchange_favorites");
   417→        if (stored) setFavorites(JSON.parse(stored));
   418→      } catch {}
   419→    };
   420→    window.addEventListener("exchange:favorites-changed", syncFavorites);
   421→    return () =>
   422→      window.removeEventListener("exchange:favorites-changed", syncFavorites);
   423→  }, []);
   424→
   425→  /* ─────── Persist favorites ─────── */
   426→  useEffect(() => {
   427→    try {
   428→      localStorage.setItem("exchange_favorites", JSON.stringify(favorites));
   429→      window.dispatchEvent(new CustomEvent("exchange:favorites-changed"));
   430→    } catch {}
   431→  }, [favorites]);
   432→
   433→  /* ─────── Toggle favorite pair ─────── */
   434→  const toggleFavorite = useCallback((pair: string) => {
   435→    setFavorites((prev) =>
   436→      prev.includes(pair)
   437→        ? prev.filter((p) => p !== pair)
   438→        : [...prev, pair]
   439→    );
   440→  }, []);
   441→
   442→  /* ─────── Sync sound enabled state ─────── */
   443→  useEffect(() => {
   444→    getSoundManager().setEnabled(soundEnabled);
   445→  }, [soundEnabled]);
   446→
   447→  /* ─────── Crosshair handler ─────── */
   448→  const handleCrosshairMove = useCallback(
   449→    (data: { candle: Candle | null; x: number; y: number } | null) => {
   450→      setCrosshairData(data?.candle || null);
   451→    },
   452→    []
   453→  );
   454→
   455→  /* ─────── Order book price click → propagate to form ─────── */
   456→  const handlePriceClick = useCallback((price: number) => {
   457→    window.dispatchEvent(
   458→      new CustomEvent("exchange:set-price", { detail: price })
   459→    );
   460→  }, []);
   461→
   462→  /* ─────── Computed ─────── */
   463→  const base = selectedPair.replace("USDT", "");
   464→  const p = prices[selectedPair];
   465→  const prevPrice = prevPrices.current[selectedPair];
   466→  const baseWallet = wallets.find((w) => w.currency === base);
   467→  const quoteWallet = wallets.find((w) => w.currency === "USDT");
   468→  const openOrdersCount = orders.filter((o) => o.status === "PENDING").length;
   469→
   470→  /* ═══════════════════════════════════════════
   471→     Render
   472→     ═══════════════════════════════════════════ */
   473→  return (
   474→    <div className="flex flex-col h-[calc(100vh-4rem)] gap-1.5 animate-fade-in overflow-hidden p-1.5">
   475→      {/* ─────── Ticker Tape (top, scrolling) ─────── */}
   476→      <TickerTape
   477→        prices={prices}
   478→        onSelectPair={setSelectedPair}
   479→        selectedPair={selectedPair}
   480→      />
   481→
   482→      {/* ─────── Pair Header (price + 24h stats + quick actions) ─────── */}
   483→      <div className="flex items-center gap-1.5">
   484→        <div className="flex-1 min-w-0">
   485→          <PairHeader
   486→            selectedPair={selectedPair}
   487→            prices={prices}
   488→            prevPrice={prevPrice}
   489→            onSelectPair={setSelectedPair}
   490→            crosshairData={crosshairData}
   491→          />
   492→        </div>
   493→
   494→        {/* Quick action buttons */}
   495→        <div className="flex items-center gap-1 shrink-0">
   496→          {/* Quick pair search (Ctrl+K) */}
   497→          <button
   498→            onClick={() => setSearchOpen(true)}
   499→            className="glass-panel rounded-lg p-2 text-muted-foreground hover:text-foreground hover:bg-muted/30 transition-all"
   500→            title="بحث سريع عن زوج (Ctrl+K)"
   501→          >
   502→            <Search className="h-4 w-4" />
   503→          </button>
   504→
   505→          {/* Chart screenshot export (Ctrl+E) */}
   506→          <button
   507→            onClick={() => {
   508→              if (chartRef.current) {
   509→                chartRef.current.exportPng(
   510→                  `chart_${selectedPair}_${timeframe}.png`
   511→                );
   512→                toast.success("تم تصدير صورة الشارت");
   513→              }
   514→            }}
   515→            className="glass-panel rounded-lg p-2 text-muted-foreground hover:text-foreground hover:bg-muted/30 transition-all"
   516→            title="تصدير صورة الشارت (Ctrl+E)"
   517→          >
   518→            <Camera className="h-4 w-4" />
   519→          </button>
   520→
   521→          {/* P&L Calculator */}
   522→          <PnLCalculator
   523→            pair={selectedPair}
   524→            base={base}
   525→            ticker={p}
   526→            feeRate={
   527→              feeSchedules.find(
   528→                (f) => f.user_type === "regular" && f.order_type === "LIMIT"
   529→              )?.taker_fee ?? 0.001
   530→            }
   531→            currentPrice={p?.price}
   532→          />
   533→
   534→          {/* Advanced orders (OCO / Trailing Stop) */}
   535→          <AdvancedOrders
   536→            pair={selectedPair}
   537→            base={base}
   538→            ticker={p}
   539→            currentPrice={p?.price}
   540→            side={side}
   541→            onPlaceOCO={(params) => {
   542→              toast(
   543→                `أمر OCO: ${params.quantity} ${base} • وقف: ${params.stopPrice} • ربح: ${params.takeProfitPrice}`,
   544→                { icon: "🎯" }
   545→              );
   546→              getSoundManager().play("order_placed");
   547→            }}
   548→            onPlaceTrailingStop={(params) => {
   549→              toast(
   550→                `إيقاف متتبع: ${params.quantity} ${base} • تتبع: ${params.trailPercent}%`,
   551→                { icon: "📈" }
   552→              );
   553→              getSoundManager().play("order_placed");
   554→            }}
   555→          />
   556→
   557→          {/* Price alerts */}
   558→          <PriceAlerts
   559→            pair={selectedPair}
   560→            currentPrice={p?.price}
   561→            compact
   562→          />
   563→
