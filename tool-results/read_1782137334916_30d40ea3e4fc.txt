    16→// ──────────────────────────────────────────────────────────────────────────
    17→// MarketHub — single source of truth for all live market data.
    18→//
    19→// Maintains ONE persistent combined stream to Binance that carries every
    20→// channel we need (miniTicker, kline, trade, depth) for every supported
    21→// symbol.  Incoming messages are routed to interested client connections
    22→// based on their subscriptions.
    23→//
    24→// Clients can either:
    25→//   • Open a dedicated single-purpose socket  →  /ws/kline, /ws/trades,
    26→//     /ws/orderbook  (backwards-compatible, simple)
    27→//   • Open a single multiplexed socket        →  /ws/market
    28→//     and send {"action":"subscribe","symbols":[...],"channels":[...]}
    29→//     messages to dynamically select what they want.
    30→// ──────────────────────────────────────────────────────────────────────────
    31→
    32→// Supported Binance symbols (lowercase for stream URLs).
    33→var supportedSymbols = []string{
    34→        "btcusdt", "ethusdt", "bnbusdt", "solusdt",
    35→        "xrpusdt", "adausdt", "dogeusdt", "dotusdt",
    36→}
    37→
    38→// Supported kline intervals.
    39→var supportedIntervals = []string{"1m", "5m", "15m", "1h", "4h", "1d"}
    40→
    41→// ── Incoming Binance message shapes ───────────────────────────────────────
    42→
    43→type binanceMiniTicker struct {
    44→        Symbol       string `json:"s"`
    45→        Price        string `json:"c"`
    46→        High         string `json:"h"`
    47→        Low          string `json:"l"`
    48→        Volume       string `json:"v"`    // base volume
    49→        QuoteVolume  string `json:"q"`    // quote volume (USDT)
    50→        PriceChange  string `json:"P"`    // 24h change %
    51→}
    52→
    53→type binanceKlineEvent struct {
    54→        Symbol string `json:"s"`
    55→        Kline  struct {
    56→                StartTime       int64  `json:"t"`
    57→                EndTime         int64  `json:"T"`
    58→                Open            string `json:"o"`
    59→                High            string `json:"h"`
    60→                Low             string `json:"l"`
    61→                Close           string `json:"c"`
    62→                Volume          string `json:"v"`
    63→                QuoteVolume     string `json:"q"`
    64→                IsClosed        bool   `json:"x"`
    65→                Interval        string `json:"i"`
    66→        } `json:"k"`
    67→}
    68→
    69→type binanceTrade struct {
    70→        Symbol    string `json:"s"`
    71→        Price     string `json:"p"`
    72→        Qty       string `json:"q"`
    73→        TradeTime int64  `json:"T"`
    74→        IsBuyer   bool   `json:"m"` // true = buyer is market maker → SELL side aggressor
    75→        TradeID   int64  `json:"t"`
    76→}
    77→
    78→type binanceDepth struct {
    79→        LastUpdateID int64        `json:"lastUpdateId"`
    80→        Bids         [][2]string  `json:"bids"`
    81→        Asks         [][2]string  `json:"asks"`
    82→}
    83→
    84→type binanceCombinedMsg struct {
    85→        Stream string          `json:"stream"`
    86→        Data   json.RawMessage `json:"data"`
    87→}
    88→
    89→// ── Outgoing client message shapes (what we send to browsers) ─────────────
    90→
    91→type ClientMsg struct {
    92→        Type string      `json:"type"` // "ticker" | "kline" | "trade" | "depth" | "snapshot" | "error"
    93→        Sym  string      `json:"symbol,omitempty"`
    94→        Data interface{} `json:"data,omitempty"`
    95→        Msg  string      `json:"message,omitempty"`
    96→}

    98→// ── Per-symbol caches ─────────────────────────────────────────────────────
    99→
   100→type TickerCache struct {
   101→        Symbol     string  `json:"symbol"`
   102→        Price      float64 `json:"price"`
   103→        Change24h  float64 `json:"change_24h"`
   104→        High24h    float64 `json:"high_24h"`
   105→        Low24h     float64 `json:"low_24h"`
   106→        Volume24h  float64 `json:"volume_24h"`
   107→        QuoteVol   float64 `json:"quote_volume_24h"`
   108→        UpdatedAt  int64   `json:"ts"`
   109→}
   110→
   111→type OrderbookCache struct {
   112→        Bids       [][2]float64 `json:"bids"`
   113→        Asks       [][2]float64 `json:"asks"`
   114→        UpdatedAt  int64        `json:"ts"`
   115→}
   116→
   117→type TradeCache struct {
   118→        ID     int64   `json:"id"`
   119→        Price  float64 `json:"price"`
   120→        Qty    float64 `json:"qty"`
   121→        Side   string  `json:"side"` // "BUY" | "SELL"
   122→        Time   int64   `json:"time"`
   123→}

   125→// ── Hub state ─────────────────────────────────────────────────────────────
   126→
   127→type MarketHub struct {
   128→        mu sync.RWMutex
   129→
   130→        // Per-symbol caches (uppercase symbol keys)
   131→        tickers    map[string]*TickerCache
   132→        orderbooks map[string]*OrderbookCache
   133→        trades     map[string][]TradeCache // last N trades per symbol
   134→
   135→        // Per-symbol per-interval last kline (for snapshot bootstrapping)
   136→        klines map[string]map[string]binanceKlineEvent // sym → interval → last kline
   137→
   138→        // Client subscriptions
   139→        clients    map[*MarketClient]bool
   140→        register   chan *MarketClient
   141→        unregister chan *MarketClient
   142→
   143→        // Binance reconnect signal
   144→        binanceReconnect chan struct{}
   145→}
   146→
   147→type MarketClient struct {
   148→        conn     *gorillaws.Conn
   149→        send     chan []byte
   150→        subs     map[string]map[string]bool // symbol → set of channels ("ticker","trade","depth","kline_1m",...)
   151→        subsMu   sync.RWMutex
   152→        hub      *MarketHub
   153→        closed   bool
   154→        closeMu  sync.Mutex
   155→}
   156→
   157→const (
   158→        clientSendBufSize = 512
   159→        maxTradesPerSym   = 30
   160→)
   161→
   162→var GlobalMarketHub = &MarketHub{
   163→        tickers:          make(map[string]*TickerCache),
   164→        orderbooks:       make(map[string]*OrderbookCache),
   165→        trades:           make(map[string][]TradeCache),
   166→        klines:           make(map[string]map[string]binanceKlineEvent),
   167→        clients:          make(map[*MarketClient]bool),
   168→        register:         make(chan *MarketClient, 64),
   169→        unregister:       make(chan *MarketClient, 64),
   170→        binanceReconnect: make(chan struct{}, 1),
   171→}
   172→
   173→func init() {
   174→        go GlobalMarketHub.run()
   175→        go GlobalMarketHub.connectBinance()
   176→}

   178→// ── Hub main loop ─────────────────────────────────────────────────────────
   179→
   180→func (h *MarketHub) run() {
   181→        ticker := time.NewTicker(2 * time.Second)
   182→        defer ticker.Stop()
   183→        for {
   184→                select {
   185→                case c := <-h.register:
   186→                        h.mu.Lock()
   187→                        h.clients[c] = true
   188→                        h.mu.Unlock()
   189→                        log.Printf("[MarketHub] client connected (%d total)", len(h.clients))
   190→                        // Send bootstrap snapshots for any symbols this client is interested in
   191→                        h.sendBootstrap(c)
   192→                case c := <-h.unregister:
   193→                        h.mu.Lock()
   194→                        if _, ok := h.clients[c]; ok {
   195→                                delete(h.clients, c)
   196→                        }
   197→                        h.mu.Unlock()
   198→                        c.safeClose()
   199→                        log.Printf("[MarketHub] client disconnected (%d total)", len(h.clients))
   200→                case <-ticker.C:
   201→                        // periodic noop — placeholder for future stats
   202→                }
   203→        }
   204→}

   206→// ── Binance combined stream ───────────────────────────────────────────────
   207→
   208→func (h *MarketHub) connectBinance() {
   209→        // Build a single combined stream URL carrying every channel/symbol we need.
   210→        streams := []string{}
   211→        for _, s := range supportedSymbols {
   212→                streams = append(streams, s+"@miniTicker")
   213→                streams = append(streams, s+"@trade")
   214→                streams = append(streams, s+"@depth20@100ms")
   215→        }
   216→        for _, s := range supportedSymbols {
   217→                for _, iv := range supportedIntervals {
   218→                        streams = append(streams, s+"@kline_"+iv)
   219→                }
   220→        }
   221→        url := "wss://stream.binance.com:9443/stream?streams=" + strings.Join(streams, "/")
   222→
   223→        for {
   224→                log.Printf("[MarketHub] connecting to Binance combined stream (%d sub-streams)", len(streams))
   225→                conn, _, err := gorillaws.DefaultDialer.Dial(url, nil)
   226→                if err != nil {
   227→                        log.Printf("[MarketHub] Binance dial failed: %v — retrying in 5s", err)
   228→                        time.Sleep(5 * time.Second)
   229→                        continue
   230→                }
   231→                log.Printf("[MarketHub] Binance combined stream connected")
   232→
   233→                for {
   234→                        _, raw, err := conn.ReadMessage()
   235→                        if err != nil {
   236→                                log.Printf("[MarketHub] Binance read error: %v — reconnecting", err)
   237→                                conn.Close()
   238→                                break
   239→                        }
   240→                        h.handleBinanceMessage(raw)
   241→                }
   242→        }
   243→}
   244→
   245→func (h *MarketHub) handleBinanceMessage(raw []byte) {
   246→        var combined binanceCombinedMsg
   247→        if err := json.Unmarshal(raw, &combined); err != nil {
   248→                return
   249→        }
   250→        parts := strings.Split(combined.Stream, "@")
   251→        if len(parts) < 2 {
   252→                return
   253→        }
   254→        symLower := parts[0]
   255→        channel := parts[1]
   256→        symUpper := strings.ToUpper(symLower)
   257→
   258→        switch {
   259→        case channel == "miniticker":
   260→                var t binanceMiniTicker
   261→                if err := json.Unmarshal(combined.Data, &t); err != nil {
   262→                        return
   263→                }
   264→                h.cacheTicker(symUpper, &t)
   265→                h.broadcastTicker(symUpper)
   266→        case strings.HasPrefix(channel, "kline_"):
   267→                interval := strings.TrimPrefix(channel, "kline_")
   268→                var k binanceKlineEvent
   269→                if err := json.Unmarshal(combined.Data, &k); err != nil {
   270→                        return
   271→                }
   272→                h.cacheKline(symUpper, interval, &k)
   273→                h.broadcastKline(symUpper, interval, &k)
   274→        case channel == "trade":
   275→                var tr binanceTrade
   276→                if err := json.Unmarshal(combined.Data, &tr); err != nil {
   277→                        return
   278→                }
   279→                h.cacheTrade(symUpper, &tr)
   280→                h.broadcastTrade(symUpper, &tr)
   281→        case strings.HasPrefix(channel, "depth"):
   282→                // Binance partial book depth stream payload (top-20 levels @ 100ms)
   283→                var d binanceDepth
   284→                if err := json.Unmarshal(combined.Data, &d); err != nil {
   285→                        return
   286→                }
   287→                h.cacheOrderbook(symUpper, &d)
   288→                h.broadcastDepth(symUpper, &d)
   289→        }
   290→}

   294→func parseFloat(s string) float64 {
   295→        f, _ := strconv.ParseFloat(s, 64)
   296→        return f
   297→}

   299→func (h *MarketHub) cacheTicker(sym string, t *binanceMiniTicker) {
   300→        h.mu.Lock()
   301→        defer h.mu.Unlock()
   302→        h.tickers[sym] = &TickerCache{
   303→                Symbol:    sym,
   304→                Price:     parseFloat(t.Price),
   305→                Change24h: parseFloat(t.PriceChange),
   306→                High24h:   parseFloat(t.High),
   307→                Low24h:    parseFloat(t.Low),
   308→                Volume24h: parseFloat(t.Volume),
   309→                QuoteVol:  parseFloat(t.QuoteVolume),
   310→                UpdatedAt: time.Now().UnixMilli(),
   311→        }
   312→}
   313→
   314→func (h *MarketHub) cacheKline(sym, interval string, k *binanceKlineEvent) {
   315→        h.mu.Lock()
   316→        defer h.mu.Unlock()
   317→        if h.klines[sym] == nil {
   318→                h.klines[sym] = make(map[string]binanceKlineEvent)
   319→        }
   320→        h.klines[sym][interval] = *k
   321→}
   322→
   323→func (h *MarketHub) cacheTrade(sym string, tr *binanceTrade) {
   324→        h.mu.Lock()
   325→        defer h.mu.Unlock()
   326→        side := "BUY"
   327→        if tr.IsBuyer {
   328→                side = "SELL" // buyer is maker → aggressor is seller
   329→        }
   330→        t := TradeCache{
   331→                ID:    tr.TradeID,
   332→                Price: parseFloat(tr.Price),
   333→                Qty:   parseFloat(tr.Qty),
   334→                Side:  side,
   335→                Time:  tr.TradeTime,
   336→        }
   337→        trades := h.trades[sym]
   338→        trades = append([]TradeCache{t}, trades...)
   339→        if len(trades) > maxTradesPerSym {
   340→                trades = trades[:maxTradesPerSym]
   341→        }
   342→        h.trades[sym] = trades
   343→}

   345→func (h *MarketHub) cacheOrderbook(sym string, d *binanceDepth) {
   346→        h.mu.Lock()
   347→        defer h.mu.Unlock()
   348→        bids := make([][2]float64, 0, len(d.Bids))
   349→        for _, b := range d.Bids {
   350→                bids = append(bids, [2]float64{parseFloat(b[0]), parseFloat(b[1])})
   351→        }
   352→        asks := make([][2]float64, 0, len(d.Asks))
   353→        for _, a := range d.Asks {
   354→                asks = append(asks, [2]float64{parseFloat(a[0]), parseFloat(a[1])})
   355→        }
   356→        h.orderbooks[sym] = &OrderbookCache{
   357→                Bids:      bids,
   358→                Asks:      asks,
   359→                UpdatedAt: time.Now().UnixMilli(),
   360→        }
   361→}

   365→func (h *MarketHub) broadcastTicker(sym string) {
   366→        h.mu.RLock()
   367→        tc, ok := h.tickers[sym]
   368→        clients := make([]*MarketClient, 0, len(h.clients))
   369→        for c := range h.clients {
   370→                clients = append(clients, c)
   371→        }
   372→        h.mu.RUnlock()
   373→        if !ok {
   374→                return
   375→        }
   376→        msg, _ := json.Marshal(ClientMsg{
   377→                Type: "ticker", Sym: sym, Data: tc,
   378→        })
   379→        for _, c := range clients {
   380→                if c.isSubscribed(sym, "ticker") {
   381→                        c.trySend(msg)
   382→                }
   383→        }
   384→}

   386→func (h *MarketHub) broadcastKline(sym, interval string, k *binanceKlineEvent) {
   387→        h.mu.RLock()
   388→        clients := make([]*MarketClient, 0, len(h.clients))
   389→        for c := range h.clients {
   390→                clients = append(clients, c)
   391→        }
   392→        h.mu.RUnlock()
   393→        out := map[string]interface{}{
   394→                "symbol":    sym,
   395→                "interval":  interval,
   396→                "time":      k.Kline.StartTime / 1000,
   397→                "open":      parseFloat(k.Kline.Open),
   398→                "high":      parseFloat(k.Kline.High),
   399→                "low":       parseFloat(k.Kline.Low),
   400→                "close":     parseFloat(k.Kline.Close),
   401→                "volume":    parseFloat(k.Kline.Volume),
   402→                "closed":    k.Kline.IsClosed,
   403→        }
   404→        msg, _ := json.Marshal(ClientMsg{
   405→                Type: "kline", Sym: sym, Data: out,
   406→        })
   407→        ch := "kline_" + interval
   408→        for _, c := range clients {
   409→                if c.isSubscribed(sym, ch) {
   410→                        c.trySend(msg)
   411→                }
   412→        }
   413→}

   475→// ── Bootstrap snapshots (sent on client connect) ──────────────────────────
   476→
   477→func (h *MarketHub) sendBootstrap(c *MarketClient) {
   478→        c.subsMu.RLock()
   479→        symbols := make([]string, 0, len(c.subs))
   480→        for sym := range c.subs {
   481→                symbols = append(symbols, sym)
   482→        }
   483→        c.subsMu.RUnlock()
   484→
   485→        if len(symbols) == 0 {
   486→                return
   487→        }
   488→
   489→        h.mu.RLock()
   490→        defer h.mu.RUnlock()
   491→
   492→        for _, sym := range symbols {
   493→                if tc, ok := h.tickers[sym]; ok && c.isSubscribed(sym, "ticker") {
   494→                        msg, _ := json.Marshal(ClientMsg{Type: "ticker", Sym: sym, Data: tc})
   495→                        c.trySend(msg)
   496→                }
   497→                if c.isSubscribed(sym, "depth") {
   498→                        if ob, ok := h.orderbooks[sym]; ok {
   499→                                msg, _ := json.Marshal(ClientMsg{
   500→                                        Type: "depth", Sym: sym,
   501→                                        Data: map[string]interface{}{
   502→                                                "symbol": sym,
   503→                                                "bids":   ob.Bids,
   504→                                                "asks":   ob.Asks,
   505→                                                "ts":     ob.UpdatedAt,
   506→                                        },
   507→                                })
   508→                                c.trySend(msg)
   509→                        }
   510→                }
   511→                if c.isSubscribed(sym, "trade") {
   512→                        if trades, ok := h.trades[sym]; ok && len(trades) > 0 {
   513→                                msg, _ := json.Marshal(ClientMsg{
   514→                                        Type: "trades_snapshot", Sym: sym,
   515→                                        Data: trades,
   516→                                })
   517→                                c.trySend(msg)
   518→                        }
   519→                }
   520→        }
   521→}

   565→// ── Client helpers ────────────────────────────────────────────────────────
   566→
   567→func (c *MarketClient) isSubscribed(sym, channel string) bool {
   568→        c.subsMu.RLock()
   569→        defer c.subsMu.RUnlock()
   570→        chans, ok := c.subs[strings.ToUpper(sym)]
   571→        if !ok {
   572→                return false
   573→        }
   574→        return chans[channel]
   575→}

   577→func (c *MarketClient) trySend(msg []byte) {
   578→        defer func() {
   579→                _ = recover()
   580→        }()
   581→        select {
   582→        case c.send <- msg:
   583→        default:
   584→                // buffer full — drop, don't block the hub
   585→        }
   586→}
   587→
   588→func (c *MarketClient) safeClose() {
   589→        c.closeMu.Lock()
   590→        defer c.closeMu.Unlock()
   591→        if c.closed {
   592→                return
   593→        }
   594→        c.closed = true
   595→        close(c.send)
   596→        _ = c.conn.Close()
   597→}

   599→// ── Subscription message handling ─────────────────────────────────────────
   600→
   601→type subAction struct {
   602→        Action   string   `json:"action"`   // "subscribe" | "unsubscribe"
   603→        Symbols  []string `json:"symbols"`
   604→        Channels []string `json:"channels"` // "ticker","trade","depth","kline_1m","kline_5m",...
   605→}
   606→
   607→func (c *MarketClient) handleSubAction(payload []byte) {
   608→        var a subAction
   609→        if err := json.Unmarshal(payload, &a); err != nil {
   610→                return
   611→        }
   612→        c.subsMu.Lock()
   613→        for _, raw := range a.Symbols {
   614→                sym := strings.ToUpper(raw)
   615→                if a.Action == "subscribe" || a.Action == "sub" {
   616→                        if c.subs[sym] == nil {
   617→                                c.subs[sym] = make(map[string]bool)
   618→                        }
   619→                        for _, ch := range a.Channels {
   620→                                c.subs[sym][ch] = true
   621→                        }
   622→                } else if a.Action == "unsubscribe" || a.Action == "unsub" {
   623→                        if c.subs[sym] == nil {
   624→                                continue
   625→                        }
   626→                        if len(a.Channels) == 0 {
   627→                                delete(c.subs, sym)
   628→                        } else {
   629→                                for _, ch := range a.Channels {
   630→                                        delete(c.subs[sym], ch)
   631→                                }
   632→                                if len(c.subs[sym]) == 0 {
   633→                                        delete(c.subs, sym)
   634→                                }
   635→                        }
   636→                }
   637→        }
   638→        c.subsMu.Unlock()
   639→        // Send bootstrap for newly-subscribed symbols
   640→        if a.Action == "subscribe" || a.Action == "sub" {
   641→                c.hub.sendBootstrap(c)
   642→        }
   643→}

   647→// HandleMarketWebSocket — multiplexed market data socket.
   648→// Clients may pass ?symbols=BTCUSDT,ETHUSDT&channels=ticker,trade,depth,kline_1m
   649→// for an initial subscription, then send JSON messages to dynamically update.
   650→func HandleMarketWebSocket(c *gin.Context) {
   651→        conn, err := upgrader.Upgrade(c.Writer, c.Request, nil)
   652→        if err != nil {
   653→                log.Printf("[MarketHub] upgrade error: %v", err)
   654→                return
   655→        }
   656→
   657→        client := &MarketClient{
   658→                conn: conn,
   659→                send: make(chan []byte, clientSendBufSize),
   660→                subs: make(map[string]map[string]bool),
   661→                hub:  GlobalMarketHub,
   662→        }
   663→
   664→        // Parse initial subscription from query string
   665→        if syms := c.Query("symbols"); syms != "" {
   666→                channels := []string{"ticker"}
   667→                if ch := c.Query("channels"); ch != "" {
   668→                        channels = strings.Split(ch, ",")
   669→                }
   670→                client.subsMu.Lock()
   671→                for _, raw := range strings.Split(syms, ",") {
   672→                        sym := strings.ToUpper(strings.TrimSpace(raw))
   673→                        if sym == "" {
   674→                                continue
   675→                        }
   676→                        if client.subs[sym] == nil {
   677→                                client.subs[sym] = make(map[string]bool)
   678→                        }
   679→                        for _, ch := range channels {
   680→                                client.subs[sym][strings.TrimSpace(ch)] = true
   681→                        }
   682→                }
   683→                client.subsMu.Unlock()
   684→        }
   685→
   686→        GlobalMarketHub.register <- client
   687→
   688→        go client.writePump()
   689→        go client.readPump()
   690→}

   756→// ── Client read/write pumps ───────────────────────────────────────────────
   757→
   758→func (c *MarketClient) readPump() {
   759→        defer func() {
   760→                c.hub.unregister <- c
   761→        }()
   762→        c.conn.SetReadLimit(4096)
   763→        c.conn.SetReadDeadline(time.Now().Add(120 * time.Second))
   764→        c.conn.SetPongHandler(func(string) error {
   765→                c.conn.SetReadDeadline(time.Now().Add(120 * time.Second))
   766→                return nil
   767→        })
   768→        for {
   769→                _, payload, err := c.conn.ReadMessage()
   770→                if err != nil {
   771→                        return
   772→                }
   773→                // Payload may be a subscribe/unsubscribe JSON action
   774→                c.handleSubAction(payload)
   775→        }
   776→}
   777→
   778→func (c *MarketClient) writePump() {
   779→        pingTicker := time.NewTicker(30 * time.Second)
   780→        defer func() {
   781→                pingTicker.Stop()
   782→                c.safeClose()
   783→        }()
   784→        for {
   785→                select {
   786→                case msg, ok := <-c.send:
   787→                        c.conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
   788→                        if !ok {
   789→                                _ = c.conn.WriteMessage(gorillaws.CloseMessage, []byte{})
   790→                                return
   791→                        }
   792→                        if err := c.conn.WriteMessage(gorillaws.TextMessage, msg); err != nil {
   793→                                return
   794→                        }
   795→                case <-pingTicker.C:
   796→                        c.conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
   797→                        if err := c.conn.WriteMessage(gorillaws.PingMessage, nil); err != nil {
   798→                                return
   799→                        }
   800→                }
   801→        }
   802→}
