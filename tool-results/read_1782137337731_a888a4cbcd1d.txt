   180тЖТ   180тЖТfunc (h *MarketHub) run() {
   181тЖТ   181тЖТ        ticker := time.NewTicker(2 * time.Second)
   182тЖТ   182тЖТ        defer ticker.Stop()
   183тЖТ   183тЖТ        for {
   184тЖТ   184тЖТ                select {
   185тЖТ   185тЖТ                case c := <-h.register:
   186тЖТ   186тЖТ                        h.mu.Lock()
   187тЖТ   187тЖТ                        h.clients[c] = true
   188тЖТ   188тЖТ                        h.mu.Unlock()
   189тЖТ   189тЖТ                        log.Printf("[MarketHub] client connected (%d total)", len(h.clients))
   190тЖТ   190тЖТ                        // Send bootstrap snapshots for any symbols this client is interested in
   191тЖТ   191тЖТ                        h.sendBootstrap(c)
   192тЖТ   192тЖТ                case c := <-h.unregister:
   193тЖТ   193тЖТ                        h.mu.Lock()
   194тЖТ   194тЖТ                        if _, ok := h.clients[c]; ok {
   195тЖТ   195тЖТ                                delete(h.clients, c)
   196тЖТ   196тЖТ                        }
   197тЖТ   197тЖТ                        h.mu.Unlock()
   198тЖТ   198тЖТ                        c.safeClose()
   199тЖТ   199тЖТ                        log.Printf("[MarketHub] client disconnected (%d total)", len(h.clients))
   200тЖТ   200тЖТ                case <-ticker.C:
   201тЖТ   201тЖТ                        // periodic noop тАФ placeholder for future stats
   202тЖТ   202тЖТ                }
   203тЖТ   203тЖТ        }
   204тЖТ   204тЖТ}

   208тЖТ   208тЖТfunc (h *MarketHub) connectBinance() {
   209тЖТ   209тЖТ        // Build a single combined stream URL carrying every channel/symbol we need.
   210тЖТ   210тЖТ        streams := []string{}
   211тЖТ   211тЖТ        for _, s := range supportedSymbols {
   212тЖТ   212тЖТ                streams = append(streams, s+"@miniTicker")
   213тЖТ   213тЖТ                streams = append(streams, s+"@trade")
   214тЖТ   214тЖТ                streams = append(streams, s+"@depth20@100ms")
   215тЖТ   215тЖТ        }
   216тЖТ   216тЖТ        for _, s := range supportedSymbols {
   217тЖТ   217тЖТ                for _, iv := range supportedIntervals {
   218тЖТ   218тЖТ                        streams = append(streams, s+"@kline_"+iv)
   219тЖТ   219тЖТ                }
   220тЖТ   220тЖТ        }
   221тЖТ   221тЖТ        url := "wss://stream.binance.com:9443/stream?streams=" + strings.Join(streams, "/")
   222тЖТ   222тЖТ
   223тЖТ   223тЖТ        for {
   224тЖТ   224тЖТ                log.Printf("[MarketHub] connecting to Binance combined stream (%d sub-streams)", len(streams))
   225тЖТ   225тЖТ                conn, _, err := gorillaws.DefaultDialer.Dial(url, nil)
   226тЖТ   226тЖТ                if err != nil {
   227тЖТ   227тЖТ                        log.Printf("[MarketHub] Binance dial failed: %v тАФ retrying in 5s", err)
   228тЖТ   228тЖТ                        time.Sleep(5 * time.Second)
   229тЖТ   229тЖТ                        continue
   230тЖТ   230тЖТ                }
   231тЖТ   231тЖТ                log.Printf("[MarketHub] Binance combined stream connected")
   232тЖТ   232тЖТ
   233тЖТ   233тЖТ                for {
   234тЖТ   234тЖТ                        _, raw, err := conn.ReadMessage()
   235тЖТ   235тЖТ                        if err != nil {
   236тЖТ   236тЖТ                                log.Printf("[MarketHub] Binance read error: %v тАФ reconnecting", err)
   237тЖТ   237тЖТ                                conn.Close()
   238тЖТ   238тЖТ                                break
   239тЖТ   239тЖТ                        }
   240тЖТ   240тЖТ                        h.handleBinanceMessage(raw)
   241тЖТ   241тЖТ                }
   242тЖТ   242тЖТ        }
   243тЖТ   243тЖТ}
   244тЖТ   244тЖТ
   245тЖТ   245тЖТfunc (h *MarketHub) handleBinanceMessage(raw []byte) {
   246тЖТ   246тЖТ        var combined binanceCombinedMsg
   247тЖТ   247тЖТ        if err := json.Unmarshal(raw, &combined); err != nil {
   248тЖТ   248тЖТ                return
   249тЖТ   249тЖТ        }
   250тЖТ   250тЖТ        parts := strings.Split(combined.Stream, "@")
   251тЖТ   251тЖТ        if len(parts) < 2 {
   252тЖТ   252тЖТ                return
   253тЖТ   253тЖТ        }
   254тЖТ   254тЖТ        symLower := parts[0]
   255тЖТ   255тЖТ        channel := parts[1]
   256тЖТ   256тЖТ        symUpper := strings.ToUpper(symLower)
   257тЖТ   257тЖТ
   258тЖТ   258тЖТ        switch {
   259тЖТ   259тЖТ        case channel == "miniticker":
   260тЖТ   260тЖТ                var t binanceMiniTicker
   261тЖТ   261тЖТ                if err := json.Unmarshal(combined.Data, &t); err != nil {
   262тЖТ   262тЖТ                        return
   263тЖТ   263тЖТ                }
   264тЖТ   264тЖТ                h.cacheTicker(symUpper, &t)
   265тЖТ   265тЖТ                h.broadcastTicker(symUpper)
   266тЖТ   266тЖТ        case strings.HasPrefix(channel, "kline_"):
   267тЖТ   267тЖТ                interval := strings.TrimPrefix(channel, "kline_")
   268тЖТ   268тЖТ                var k binanceKlineEvent
   269тЖТ   269тЖТ                if err := json.Unmarshal(combined.Data, &k); err != nil {
   270тЖТ   270тЖТ                        return
   271тЖТ   271тЖТ                }
   272тЖТ   272тЖТ                h.cacheKline(symUpper, interval, &k)
   273тЖТ   273тЖТ                h.broadcastKline(symUpper, interval, &k)
   274тЖТ   274тЖТ        case channel == "trade":
   275тЖТ   275тЖТ                var tr binanceTrade
   276тЖТ   276тЖТ                if err := json.Unmarshal(combined.Data, &tr); err != nil {
   277тЖТ   277тЖТ                        return
   278тЖТ   278тЖТ                }
   279тЖТ   279тЖТ                h.cacheTrade(symUpper, &tr)
   280тЖТ   280тЖТ                h.broadcastTrade(symUpper, &tr)
   281тЖТ   281тЖТ        case strings.HasPrefix(channel, "depth"):
   282тЖТ   282тЖТ                // Binance partial book depth stream payload (top-20 levels @ 100ms)
   283тЖТ   283тЖТ                var d binanceDepth
   284тЖТ   284тЖТ                if err := json.Unmarshal(combined.Data, &d); err != nil {

   599тЖТ   599тЖТ// тФАтФА Subscription message handling тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   600тЖТ   600тЖТ
   601тЖТ   601тЖТtype subAction struct {
   602тЖТ   602тЖТ        Action   string   `json:"action"`   // "subscribe" | "unsubscribe"
   603тЖТ   603тЖТ        Symbols  []string `json:"symbols"`
   604тЖТ   604тЖТ        Channels []string `json:"channels"` // "ticker","trade","depth","kline_1m","kline_5m",...
   605тЖТ   605тЖТ}
   606тЖТ   606тЖТ
   607тЖТ   607тЖТfunc (c *MarketClient) handleSubAction(payload []byte) {
   608тЖТ   608тЖТ        var a subAction
   609тЖТ   609тЖТ        if err := json.Unmarshal(payload, &a); err != nil {
   610тЖТ   610тЖТ                return
   611тЖТ   611тЖТ        }
   612тЖТ   612тЖТ        c.subsMu.Lock()
   613тЖТ   613тЖТ        for _, raw := range a.Symbols {
   614тЖТ   614тЖТ                sym := strings.ToUpper(raw)
   615тЖТ   615тЖТ                if a.Action == "subscribe" || a.Action == "sub" {
   616тЖТ   616тЖТ                        if c.subs[sym] == nil {
   617тЖТ   617тЖТ                                c.subs[sym] = make(map[string]bool)
   618тЖТ   618тЖТ                        }
   619тЖТ   619тЖТ                        for _, ch := range a.Channels {
   620тЖТ   620тЖТ                                c.subs[sym][ch] = true
   621тЖТ   621тЖТ                        }
   622тЖТ   622тЖТ                } else if a.Action == "unsubscribe" || a.Action == "unsub" {
   623тЖТ   623тЖТ                        if c.subs[sym] == nil {
   624тЖТ   624тЖТ                                continue
   625тЖТ   625тЖТ                        }
   626тЖТ   626тЖТ                        if len(a.Channels) == 0 {
   627тЖТ   627тЖТ                                delete(c.subs, sym)
   628тЖТ   628тЖТ                        } else {
   629тЖТ   629тЖТ                                for _, ch := range a.Channels {
   630тЖТ   630тЖТ                                        delete(c.subs[sym], ch)
   631тЖТ   631тЖТ                                }
   632тЖТ   632тЖТ                                if len(c.subs[sym]) == 0 {
   633тЖТ   633тЖТ                                        delete(c.subs, sym)
   634тЖТ   634тЖТ                                }
   635тЖТ   635тЖТ                        }
   636тЖТ   636тЖТ                }
   637тЖТ   637тЖТ        }
   638тЖТ   638тЖТ        c.subsMu.Unlock()
   639тЖТ   639тЖТ        // Send bootstrap for newly-subscribed symbols
   640тЖТ   640тЖТ        if a.Action == "subscribe" || a.Action == "sub" {
   641тЖТ   641тЖТ                c.hub.sendBootstrap(c)
   642тЖТ   642тЖТ        }
   643тЖТ   643тЖТ}

   647тЖТ   647тЖТ// HandleMarketWebSocket тАФ multiplexed market data socket.
   648тЖТ   648тЖТ// Clients may pass ?symbols=BTCUSDT,ETHUSDT&channels=ticker,trade,depth,kline_1m
   649тЖТ   649тЖТ// for an initial subscription, then send JSON messages to dynamically update.
   650тЖТ   650тЖТfunc HandleMarketWebSocket(c *gin.Context) {
   651тЖТ   651тЖТ        conn, err := upgrader.Upgrade(c.Writer, c.Request, nil)
   652тЖТ   652тЖТ        if err != nil {
   653тЖТ   653тЖТ                log.Printf("[MarketHub] upgrade error: %v", err)
   654тЖТ   654тЖТ                return
   655тЖТ   655тЖТ        }
   656тЖТ   656тЖТ
   657тЖТ   657тЖТ        client := &MarketClient{
   658тЖТ   658тЖТ                conn: conn,
   659тЖТ   659тЖТ                send: make(chan []byte, clientSendBufSize),
   660тЖТ   660тЖТ                subs: make(map[string]map[string]bool),
   661тЖТ   661тЖТ                hub:  GlobalMarketHub,
   662тЖТ   662тЖТ        }
   663тЖТ   663тЖТ
   664тЖТ   664тЖТ        // Parse initial subscription from query string
   665тЖТ   665тЖТ        if syms := c.Query("symbols"); syms != "" {
   666тЖТ   666тЖТ                channels := []string{"ticker"}
   667тЖТ   667тЖТ                if ch := c.Query("channels"); ch != "" {
   668тЖТ   668тЖТ                        channels = strings.Split(ch, ",")
   669тЖТ   669тЖТ                }
   670тЖТ   670тЖТ                client.subsMu.Lock()
   671тЖТ   671тЖТ                for _, raw := range strings.Split(syms, ",") {
   672тЖТ   672тЖТ                        sym := strings.ToUpper(strings.TrimSpace(raw))
   673тЖТ   673тЖТ                        if sym == "" {
   674тЖТ   674тЖТ                                continue
   675тЖТ   675тЖТ                        }
   676тЖТ   676тЖТ                        if client.subs[sym] == nil {
   677тЖТ   677тЖТ                                client.subs[sym] = make(map[string]bool)
   678тЖТ   678тЖТ                        }
   679тЖТ   679тЖТ                        for _, ch := range channels {
   680тЖТ   680тЖТ                                client.subs[sym][strings.TrimSpace(ch)] = true
   681тЖТ   681тЖТ                        }
   682тЖТ   682тЖТ                }
   683тЖТ   683тЖТ                client.subsMu.Unlock()
   684тЖТ   684тЖТ        }
   685тЖТ   685тЖТ
   686тЖТ   686тЖТ        GlobalMarketHub.register <- client
   687тЖТ   687тЖТ
   688тЖТ   688тЖТ        go client.writePump()
   689тЖТ   689тЖТ        go client.readPump()
   690тЖТ   690тЖТ}

   758тЖТ   758тЖТfunc (c *MarketClient) readPump() {
   759тЖТ   759тЖТ        defer func() {
   760тЖТ   760тЖТ                c.hub.unregister <- c
   761тЖТ   761тЖТ        }()
   762тЖТ   762тЖТ        c.conn.SetReadLimit(4096)
   763тЖТ   763тЖТ        c.conn.SetReadDeadline(time.Now().Add(120 * time.Second))
   764тЖТ   764тЖТ        c.conn.SetPongHandler(func(string) error {
   765тЖТ   765тЖТ                c.conn.SetReadDeadline(time.Now().Add(120 * time.Second))
   766тЖТ   766тЖТ                return nil
   767тЖТ   767тЖТ        })
   768тЖТ   768тЖТ        for {
   769тЖТ   769тЖТ                _, payload, err := c.conn.ReadMessage()
   770тЖТ   770тЖТ                if err != nil {
   771тЖТ   771тЖТ                        return
   772тЖТ   772тЖТ                }
   773тЖТ   773тЖТ                // Payload may be a subscribe/unsubscribe JSON action
   774тЖТ   774тЖТ                c.handleSubAction(payload)
   775тЖТ   775тЖТ        }
   776тЖТ   776тЖТ}

   778тЖТ   778тЖТfunc (c *MarketClient) writePump() {
   779тЖТ   779тЖТ        pingTicker := time.NewTicker(30 * time.Second)
   780тЖТ   780тЖТ        defer func() {
   781тЖТ   781тЖТ                pingTicker.Stop()
   782тЖТ   782тЖТ                c.safeClose()
   783тЖТ   783тЖТ        }()
   784тЖТ   784тЖТ        for {
   785тЖТ   785тЖТ                select {
   786тЖТ   786тЖТ                case msg, ok := <-c.send:
   787тЖТ   787тЖТ                        c.conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
   788тЖТ   788тЖТ                        if !ok {
   789тЖТ   789тЖТ                                _ = c.conn.WriteMessage(gorillaws.CloseMessage, []byte{})
   790тЖТ   790тЖТ                                return
   791тЖТ   791тЖТ                        }
   792тЖТ   792тЖТ                        if err := c.conn.WriteMessage(gorillaws.TextMessage, msg); err != nil {
   793тЖТ   793тЖТ                                return
   794тЖТ   794тЖТ                        }
   795тЖТ   795тЖТ                case <-pingTicker.C:
   796тЖТ   796тЖТ                        c.conn.SetWriteDeadline(time.Now().Add(10 * time.Second))
   797тЖТ   797тЖТ                        if err := c.conn.WriteMessage(gorillaws.PingMessage, nil); err != nil {
   798тЖТ   798тЖТ                                return
   799тЖТ   799тЖТ                        }
   800тЖТ   800тЖТ                }
   801тЖТ   801тЖТ        }
   802тЖТ   802тЖТ}
